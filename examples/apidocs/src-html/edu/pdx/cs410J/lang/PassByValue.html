001package edu.pdx.cs410J.lang;
002
003/**
004 * Demonstrates Java's pass by value mechanism
005 */
006public class PassByValue {
007  private static void doubled(int i) {
008    i = i * 2;
009    System.out.println("Doubled: " + i);
010  }
011
012  public static void main(String[] args) {
013    int i = 27;
014    System.out.println("Before: " + i);
015    doubled(i);
016    System.out.println("After: " + i);
017  }
018}
