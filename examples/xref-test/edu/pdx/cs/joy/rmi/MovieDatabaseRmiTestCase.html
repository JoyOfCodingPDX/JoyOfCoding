4   import org.junit.jupiter.api.AfterAll;

7   import java.net.MalformedURLException;
8   import java.rmi.AlreadyBoundException;
9   import java.rmi.NotBoundException;
10  import java.rmi.RemoteException;
11  import java.rmi.registry.LocateRegistry;
12  import java.rmi.registry.Registry;
13  import java.rmi.server.UnicastRemoteObject;
14  
15  public class MovieDatabaseRmiTestCase extends InvokeMainTestCase {
16    static int rmiPort;
17  
18    private static int getRandomPort() {
19      return (int) (1_025 + Math.random() * (9_000 - 1));
20    }
21  
22    static final String RMI_HOST = "localhost";
23    private static MovieDatabaseImpl movieDatabaseImpl;
24    private static Registry rmiRegistry;
25  
26    @BeforeAll
27    public static void bindMovieDatabaseIntoRmiRegistry() throws RemoteException, AlreadyBoundException, MalformedURLException {
28      movieDatabaseImpl = new MovieDatabaseImpl();
29      rmiPort = getRandomPort();
30      MovieDatabase database = (MovieDatabase) UnicastRemoteObject.exportObject(movieDatabaseImpl, rmiPort);
31      rmiRegistry = LocateRegistry.createRegistry(rmiPort);
32      rmiRegistry.bind(MovieDatabase.RMI_OBJECT_NAME, database);
33    }
34  
35    @AfterAll
36    public static void unbindMovieDatabaseFromRmiRegistry() throws RemoteException, NotBoundException {
37      Registry registry = rmiRegistry;
38      UnicastRemoteObject.unexportObject(movieDatabaseImpl, true);
39      registry.unbind(MovieDatabase.RMI_OBJECT_NAME);
40      UnicastRemoteObject.unexportObject(rmiRegistry, true);
41      movieDatabaseImpl = null;
42    }
43  
44    private static Registry getRmiRegistry() throws RemoteException {
45      return LocateRegistry.getRegistry(RMI_HOST, rmiPort);
46    }
47  
48    static MovieDatabase getMovieDatabase() throws RemoteException, NotBoundException {
49      return (MovieDatabase) getRmiRegistry().lookup(MovieDatabase.RMI_OBJECT_NAME);
50    }
51  }
