3   import java.io.*;
4   
5   /**
6    * This program reads text from the console until the user enters
7    * <code>-1</code> at which point it prints what the user has entered.
8    * It demonstrates the <code>BufferedReader</code>,
9    * <code>InputStreamReader</code>, and <code>StringWriter</code>
10   * classes.
11   */
12  public class ReadFromConsole {
13  
14    public static void main(String[] args) {
15      InputStreamReader isr = new InputStreamReader(System.in);
16      BufferedReader br = new BufferedReader(isr);
17      StringWriter text = new StringWriter();
18  
19      while (true) {
20        try {
21  	// Read a line from the console
22  	String line = br.readLine();
23  
24  	if(line.equals("-1")) {
25  	  // All done with input
26  	  break;
27  
28  	} else {
29  	  text.write(line + " ");
30  	}
31  
32        } catch (IOException ex) {
33  	System.err.println("** " + ex);
34  	System.exit(1);
35        }
36      }
37  
38      // Print out what was entered
39      System.out.println(text);
40    }
41  
42  }
