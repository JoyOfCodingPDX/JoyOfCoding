6    * Prints out information about today's date.  Demonstrates the
7    * <code>Date</code> and <code>Calendar</code> classes.

10  
11    public static void main(String[] args) {
12      Date today = new Date();
13      Calendar cal = Calendar.getInstance();
14      cal.setTime(today);
15  
16      int dayOfWeek = cal.get(Calendar.DAY_OF_WEEK);
17      int dayOfYear = cal.get(Calendar.DAY_OF_YEAR);
18      int weekOfMonth = cal.get(Calendar.WEEK_OF_MONTH);
19  
20      StringBuffer sb = new StringBuffer();
21      sb.append("Today is " + today + "\n");
22      sb.append("It's been " + today.getTime() +
23  	      "ms since the epoch.");
24      sb.append("\nIt is the " + dayOfWeek + 
25  	      "th day of the week \nand the " +
26  	      dayOfYear + "th day of the year.  ");
27      sb.append("\nWe are in the " + weekOfMonth +
28  	      "th week of the month.");
29      System.out.println(sb.toString());
30  
31    }
