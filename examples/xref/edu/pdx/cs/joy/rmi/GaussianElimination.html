3   /**
4    * This class performs Gaussian Elimination to solve a system of
5    * linear equations of the form <code>Ax = b</code>.
6    */
7   public class GaussianElimination {
8   
9     /**
10     * Solve a system <code>Ax = b</code> using Gaussian elimination
11     *
12     * @throws IllegalArgumentException
13     *         <code>A</code> is not an <code>n x n</code> matrix,
14     *         <code>b</code> does not have length <code>n</code>
15     */
16    public static double[] solve(double[][] a, double[] b) {
17      // First validate that 
18  
19      int rows = Matrix.countRows(a);
20      int columns = Matrix.countColumns(a);
21      if (rows != columns) {
22        String s = "Matrix a must have the same number of rows (" + rows
23          + ") and columns (" + columns + ")";
24        throw new IllegalArgumentException(s);
25      }
26  
27      if (b.length != rows) {
28        String s = "Vector b must have " + rows + " rows (not " +
29        b.length + ")";
30        throw new IllegalArgumentException(s);
31      }
32  
33      java.io.PrintWriter pw = new java.io.PrintWriter(System.out, true);
34      Matrix.print("Matrix A", a, pw);
35      Matrix.print("Vector b", b, pw);
36  
37      int n = rows;
38      double[] x = new double[n];
39  
40      // Transform "a" into a upper-diagnol matrix
41  
42      // Divide each element in the row by a[0][0]
43      double d = a[0][0];
44      for(int h = 0; h < n; h++) {
45        a[0][h] = a[0][h] / d;
46      }
47      b[0] = b[0] / d;
48      
49      for (int k = 0; k < n; k++) {
50        for (int i = k + 1; i < n; i++) {
51          double multiplier = a[i][k] / a[k][k];
52          for (int j = k; j < n; j++) {
53            a[i][j] = a[i][j] - (multiplier * a[k][j]);
54          }
55          b[i] = b[i] - (multiplier * b[k]);
56        }
57      }
58  
59      // Now solve for x from the bottom up
60      for (int i = n-1; i >= 0; i--) {
61        double v = b[i];
62        for (int j = i+1; j < n; j++) {
63          v -= (a[i][j] * x[j]);
64        }
65        x[i] = v / a[i][i];
66      }
67  
68      Matrix.print("Vector x", x, pw);
69      
70      return x;
71    }
