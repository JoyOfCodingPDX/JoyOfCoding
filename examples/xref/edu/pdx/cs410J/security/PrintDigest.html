3   import java.security.*;

5   /**
6    * This program computes the MD5 digest of a sentence specified on the
7    * command line.
8    */
9   public class PrintDigest {
10  
11    public static void main(String[] args) {
12      String message = args[0];
13      MessageDigest algorithm = null;
14      try {
15        algorithm = MessageDigest.getInstance("MD5");
16      } catch (NoSuchAlgorithmException ex) {
17        ex.printStackTrace(System.err);
18        System.exit(1);
19      }
20  
21      algorithm.reset();
22      algorithm.update(message.getBytes());
23      byte[] digest = algorithm.digest();
24  
25      StringBuffer hexString = new StringBuffer();
26      for (int i = 0; i < digest.length; i++) {
27        String s = Integer.toHexString(0xFF & digest[i]);
28        hexString.append(s);
29      }
30      
31      System.out.println("Message: " + message);
32      System.out.println("Digest: " + hexString.toString());
33    }
34  }
