3   import java.io.PrintStream;
4   import java.util.Calendar;
5   
6   /**
7    * Demonstrated J2SE 1.5's facilities for <code>printf</code>-style
8    * formatting. 
9    *
10   * @see java.util.Formatter
11   *
12   * @author David Whitlock
13   * @version $Revision: 1.2 $
14   * @since Summer 2004
15   */
16  public class Formatting {
17  
18    /**
19     * Formats a number of different kinds of data using J2SE 1.5's
20     * formatting facilities.
21     */
22    public static void main(String[] args) {
23      PrintStream out = System.out;
24      out.printf("%s%n", "Hello World");
25  
26      Calendar today = Calendar.getInstance();
27  
28      out.printf("Today's date is: %tm/%td/%tY%n", today, today, today);
29      out.printf("The current time is: %tl:%tM %tp%n", today, today, today);
30  
31      out.printf("%f/%.2f = %f%n", 2.0, 3.0, (2.0/3.0));
32  
33      for (int i = 0; i < 3; i++) {
34        out.printf("%5s%5s%5s%n", i, i+1, i+2);
35      }
36  
37      out.printf("%-10s%s%n", "left", "right");
38    }
39  
40  }
