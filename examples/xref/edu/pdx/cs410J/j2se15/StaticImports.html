15    /**
16     * Prints the sum of the integers entered on the command line
17     */
18    public static void main(String[] args) {
19      int sum = 0;
20      for (int i = 0; i < args.length; i++) {
21        sum += parseInt(args[i]); // Integer.parseInt()
22      }
23  
24      out.println("Sum is " + sum); // System.out
25      out.println("MAX_INT is " + MAX_VALUE); // Integer.MAX_VALUE
26    }
