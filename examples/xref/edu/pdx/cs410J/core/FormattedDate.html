3   import java.text.*;
4   import java.util.*;
5   
6   /**
7    * This program reads a date and time from the command in
8    * <code>DateFormat.MEDIUM</code> format and prints it back out in all
9    * four formats.
10   */
11  public class FormattedDate {
12  
13    /**
14     * The command line contains a date to be formatted
15     */
16    public static void main(String[] args) {
17      StringBuffer sb = new StringBuffer();
18      for (int i = 0; i < args.length; i++) {
19        sb.append(args[i] + " ");
20      }
21      Date date = null;
22      DateFormat df = 
23        DateFormat.getDateTimeInstance(DateFormat.MEDIUM, DateFormat.MEDIUM);
24      try {
25        date = df.parse(sb.toString().trim());
26  
27      } catch (ParseException ex) {
28        String s = "Bad date: " + sb;
29        System.err.println("** " + s);
30        System.exit(1);
31      }
32  
33      df = DateFormat.getDateTimeInstance(DateFormat.SHORT, DateFormat.SHORT);
34      System.out.println("SHORT: " + df.format(date));
35  
36      df = DateFormat.getDateTimeInstance(DateFormat.MEDIUM, DateFormat.MEDIUM);
37      System.out.println("MEDIUM: " + df.format(date));
38  
39      df = DateFormat.getDateTimeInstance(DateFormat.LONG, DateFormat.LONG);
40      System.out.println("LONG: " + df.format(date));
41  
42      df = DateFormat.getDateTimeInstance(DateFormat.FULL, DateFormat.FULL);
43      System.out.println("FULL: " + df.format(date));
44    }
45  
46  }
