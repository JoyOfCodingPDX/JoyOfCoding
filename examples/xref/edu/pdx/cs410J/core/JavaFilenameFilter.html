5   /**
6    * This class is a <code>FilenameFilter</code> that only accepts files
7    * that end in .java.
8    */
9   public class JavaFilenameFilter implements FilenameFilter {
10  
11    /**
12     * Returns <code>true</code> if a file's name ends with ".java"
13     */
14    public boolean accept(File dir, String fileName) {
15      if (fileName.endsWith(".java")) {
16        return true;
17  
18      } else {
19        return false;
20      }
21    }
