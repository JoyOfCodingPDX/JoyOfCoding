5   /**
6    * This program demonstrates what happens when you modify a collection
7    * while iterating over it.
8    */
9   public class ModifyWhileIterating {
10  
11    public static void main(String[] args) {
12      List<String> list = new ArrayList<>();
13      list.add("one");
14      list.add("two");
15  
16      Iterator<String> iter = list.iterator();
17      while (iter.hasNext()) {
18        String s = iter.next();
19        if (s.equals("one")) {
20          list.add(0, "start");
21        }
22      }
23    }
24  
25  }
