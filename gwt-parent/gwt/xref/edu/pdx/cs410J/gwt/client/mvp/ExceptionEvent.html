1   package edu.pdx.cs410J.gwt.client.mvp;
2   
3   import com.google.gwt.event.shared.EventHandler;
4   import com.google.gwt.event.shared.GwtEvent;
5   
6   /**
7    * Event triggered when an exception is thrown, most often from an {@link MvpCallback} 
8    */
9   public class ExceptionEvent extends GwtEvent<ExceptionEvent.Handler> {
10  
11    public static Type<Handler> TYPE = new Type<Handler>();
12  
13    private final Throwable exception;
14  
15    public interface Handler extends EventHandler {
16      void onException(Throwable ex);
17    }
18  
19    public ExceptionEvent(Throwable ex) {
20      this.exception = ex;
21    }
22  
23    @Override
24    public Type<Handler> getAssociatedType() {
25      return TYPE;
26    }
27  
28    @Override
29    protected void dispatch(Handler handler) {
30      handler.onException(this.exception);
31    }
32  }
