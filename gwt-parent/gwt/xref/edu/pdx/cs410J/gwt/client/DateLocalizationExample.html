3   import com.google.gwt.core.client.GWT;
4   import com.google.gwt.event.logical.shared.ValueChangeEvent;
5   import com.google.gwt.event.logical.shared.ValueChangeHandler;
6   import com.google.gwt.i18n.client.DateTimeFormat;
7   import com.google.gwt.user.client.ui.Label;
8   import com.google.gwt.user.datepicker.client.CalendarUtil;
9   import com.google.gwt.user.datepicker.client.DateBox;
10  
11  import java.util.Date;
12  
13  public class DateLocalizationExample extends Example
14  {
15      private final Label label;
16  
17      public DateLocalizationExample()
18      {
19          super( "Date Localization" );
20  
21          DateLocalizationStrings strings = GWT.create( DateLocalizationStrings.class );
22          this.label = new Label();
23          this.label.setText( strings.defaultDaysMessage() );
24          this.add(new Label(strings.selectString()));
25  
26          final DateLocalizationMessages messages = GWT.create( DateLocalizationMessages.class );
27  
28          DateBox dateBox = new DateBox();
29          dateBox.setFormat( new DateBox.DefaultFormat( DateTimeFormat.getLongDateFormat()) );
30          Date date = new Date();
31          CalendarUtil.addMonthsToDate( date, -29 * 12 );
32          dateBox.setValue( date );
33  
34          dateBox.addValueChangeHandler( new ValueChangeHandler<Date>() {
35  
36              public void onValueChange( ValueChangeEvent<Date> e )
37              {
38                long days = CalendarUtil.getDaysBetween(e.getValue(), new Date() );
39                label.setText( messages.daysMessage( days ) );
40              }
41          });
42  
43          this.add( dateBox );
44          this.add( label );
45      }
46  }
