082    verify(view).setDividendChangeHandler(captor.capture());
083
084    int value = 123;
085    fireValueChangeEvent(String.valueOf(value), captor.getValue());

206  /**
207   * Tests that the quotient is set appropriately
208   */
209  @Test
210  public void testQuotientIsSet() {
211    DivisionPresenter.Display view = mock(DivisionPresenter.Display.class);
212    DivisionServiceAsync service = mock(DivisionServiceAsync.class);
213
214    DivisionPresenter presenter = new DivisionPresenter(view, service, eventBus);
215    ArgumentCaptor<ClickHandler> clickCaptor = ArgumentCaptor.forClass(ClickHandler.class);
216    verify(view).setDivisionClickHandler(clickCaptor.capture());
217
218    presenter.setDivisor(123);
219    presenter.setDividend(234);
220
221    fireClickEvent(clickCaptor.getValue());
222
223    verify(service).divide(anyInt(), anyInt(), callbackCaptor.capture());
224
225    Integer quotient = 456;
226    callbackCaptor.getValue().onSuccess(quotient);
227
228    verify(view).setQuotient(String.valueOf(quotient));
229    verify(view, never()).setErrorMessage(any(String.class));
230  }

251
252    final String message = "This is an error message";
253    callbackCaptor.getValue().onFailure(new RuntimeException() {
254      @Override
255      public String toString() {
256        return message;
257      }
258    });
259
260    verify(view, never()).setQuotient(any(String.class));
261    verify(view).setErrorMessage(message);
262  }

264  /**
265   * Fires a {@link ClickEvent} to a given handler
266   *
267   * @param handler The handler that wlll receive the event
268   */
269  private void fireClickEvent(ClickHandler handler) {
270    // Firing a ClickEvent requires a NativeEvent which is only available with GWT.create(), not in JUnit tests
271    handler.onClick(mock(ClickEvent.class));
272  }
273
274  /**
275   * Fires a {@link ValueChangeEvent} to a given handler
276   *
277   * @param value   The value that is changing
278   * @param handler The handler that will receive the event
279   */
280  private <T> void fireValueChangeEvent(T value, ValueChangeHandler<T> handler) {
281    HasValueChangeHandlers<T> mock = new MockHasValueChangeHandlers<T>();
282    mock.addValueChangeHandler(handler);
283    ValueChangeEvent.fire(mock, value);
284  }
285
286  private abstract class MockHasHandlers implements HasHandlers {
287    protected final HandlerManager manager;
288
289    private MockHasHandlers() {
290      manager = new HandlerManager(this);
291    }
292
293    public void fireEvent(GwtEvent<?> event) {
294      manager.fireEvent(event);
295    }
296  }
297
298  /**
299   * A mock that takes care of the handling and firing events
300   *
301   * @param <T>
302   */
303  private class MockHasValueChangeHandlers<T> extends MockHasHandlers implements HasValueChangeHandlers<T> {
304    public HandlerRegistration addValueChangeHandler(ValueChangeHandler<T> handler) {
305      return manager.addHandler(ValueChangeEvent.getType(), handler);
306    }
307
308  }
