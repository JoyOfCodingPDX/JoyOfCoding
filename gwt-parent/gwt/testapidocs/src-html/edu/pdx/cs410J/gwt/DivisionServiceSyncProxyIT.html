001package edu.pdx.cs410J.gwt;
002
003import com.gdevelop.gwt.syncrpc.SyncProxy;
004import edu.pdx.cs410J.gwt.client.DivisionService;
005import edu.pdx.cs410J.web.HttpRequestHelper;
006import org.junit.Test;
007
008import java.io.IOException;
009
010import static junit.framework.TestCase.assertEquals;
011
012/**
013 * Uses GWT sync proxy to invoke the <code>DivisionService</code>.
014 * This verifies that the GWT web application is up and running in Jetty.
015 */
016public class DivisionServiceSyncProxyIT extends HttpRequestHelper {
017
018  private final int httpPort = Integer.getInteger("http.port", 8080);
019  private String webAppUrl = "http://localhost:" + httpPort + "/gwt";
020
021  @Test
022  public void gwtWebApplicationIsRunning() throws IOException {
023    Response response = get(this.webAppUrl);
024    assertEquals(200, response.getCode());
025  }
026
027  @Test
028  public void canInvokeDivisionServiceWithGwtSyncProxy() {
029    String moduleName = "examples";
030    SyncProxy.setBaseURL(this.webAppUrl + "/" + moduleName + "/");
031
032    DivisionService sync = SyncProxy.createSync(DivisionService.class);
033    int quotient = sync.divide(6, 2);
034    assertEquals(3, quotient);
035  }
036}
