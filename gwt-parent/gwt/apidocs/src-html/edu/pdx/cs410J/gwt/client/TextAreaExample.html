001package edu.pdx.cs410J.gwt.client;
002
003import com.google.gwt.user.client.ui.TextArea;
004import com.google.gwt.user.client.ui.RichTextArea;
005
006/**
007 * Demonstrates the GWT {@link TextArea} and {@link RichTextArea}
008 */
009public class TextAreaExample extends Example {
010  public TextAreaExample() {
011    super("Text Area");
012
013    TextArea text = new TextArea();
014    text.setCharacterWidth(30);
015    text.setVisibleLines(5);
016    add(text);
017
018    RichTextArea rich = new RichTextArea();
019    rich.setHTML("Rich <i>text</i> <b>area</b>");
020    add(rich);
021  }
022}
