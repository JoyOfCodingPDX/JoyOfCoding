001package edu.pdx.cs410J.gwt.client;
002
003import com.google.gwt.event.dom.client.ChangeHandler;
004import com.google.gwt.event.dom.client.ChangeEvent;
005import com.google.gwt.user.client.ui.*;
006
007/**
008 * Demonstrates GWT's {@link DeckPanel}
009 */
010public class DeckPanelExample extends Example {
011  public DeckPanelExample() {
012    super("Deck Panel");
013
014    DockPanel dock = new DockPanel();
015    final DeckPanel deck = new DeckPanel();
016    final ListBox list = new ListBox();
017    list.addChangeHandler(new ChangeHandler() {
018        public void onChange( ChangeEvent changeEvent )
019        {
020            int index = list.getSelectedIndex();
021            deck.showWidget(index);
022        }
023    });
024
025    for (String label : new String[] { "One", "Two", "Three", "Four"}) {
026      list.addItem(label);
027      deck.add(new Button(label));
028    }
029    deck.showWidget(0);
030
031    dock.add(list, DockPanel.NORTH);
032    dock.add(deck, DockPanel.CENTER);
033    add(dock);
034  }
035}
