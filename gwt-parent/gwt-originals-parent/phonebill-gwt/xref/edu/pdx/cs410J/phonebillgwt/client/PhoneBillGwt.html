3   import com.google.common.annotations.VisibleForTesting;
4   import com.google.gwt.core.client.EntryPoint;
5   import com.google.gwt.core.client.GWT;
6   import com.google.gwt.event.dom.client.ClickEvent;
7   import com.google.gwt.event.dom.client.ClickHandler;
8   import com.google.gwt.user.client.Window;
9   import com.google.gwt.user.client.rpc.AsyncCallback;
10  import com.google.gwt.user.client.ui.Button;
11  import com.google.gwt.user.client.ui.RootPanel;
12  
13  import java.util.Collection;
14  
15  /**
16   * A basic GWT class that makes sure that we can send an Phone Bill back from the server
17   */
18  public class PhoneBillGwt implements EntryPoint {
19    private final Alerter alerter;
20  
21    @VisibleForTesting
22    Button button;
23  
24    public PhoneBillGwt() {
25      this(new Alerter() {
26        @Override
27        public void alert(String message) {
28          Window.alert(message);
29        }
30      });
31    }
32  
33    @VisibleForTesting
34    PhoneBillGwt(Alerter alerter) {
35      this.alerter = alerter;
36  
37      addWidgets();
38    }
39  
40    private void addWidgets() {
41      button = new Button("Ping Server");
42      button.addClickHandler(new ClickHandler() {
43        @Override
44        public void onClick(ClickEvent clickEvent) {
45          PingServiceAsync async = GWT.create(PingService.class);
46          async.ping(new AsyncCallback<PhoneBill>() {
47  
48            @Override
49            public void onFailure(Throwable ex) {
50              alerter.alert(ex.toString());
51            }
52  
53            @Override
54            public void onSuccess(PhoneBill bill) {
55              StringBuilder sb = new StringBuilder(bill.toString());
56              Collection<PhoneCall> calls = bill.getPhoneCalls();
57              for (PhoneCall call : calls) {
58                sb.append(call);
59                sb.append("\n");
60              }
61              alerter.alert(sb.toString());
62            }
63          });
64        }
65      });
66    }
67  
68    @Override
69    public void onModuleLoad() {
70      RootPanel rootPanel = RootPanel.get();
71      rootPanel.add(button);
72    }
73  
74    @VisibleForTesting
75    interface Alerter {
76      void alert(String message);
77    }
78  
79  }
