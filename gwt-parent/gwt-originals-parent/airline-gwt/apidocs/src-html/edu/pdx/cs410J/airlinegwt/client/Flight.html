007public class Flight extends AbstractFlight

014
015  }
016
017  @Override
018  public int getNumber() {
019    return 42;
020  }
021
022  @Override
023  public String getSource() {
024    return "PDX";
025  }
026
027  @Override
028  public Date getDeparture() {

030  }
031
032  public String getDepartureString() {
033    return "DEPART " + getDeparture();
034  }
035
036  public String getDestination() {

038  }
039
040  public Date getArrival() {

042  }
043
044  public String getArrivalString() {

047
048}
