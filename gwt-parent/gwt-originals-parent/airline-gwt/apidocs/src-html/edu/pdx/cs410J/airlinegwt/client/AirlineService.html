001package edu.pdx.cs410J.airlinegwt.client;
002
003import com.google.gwt.user.client.rpc.RemoteService;
004import com.google.gwt.user.client.rpc.RemoteServiceRelativePath;
005
006/**
007 * A GWT remote service that returns a dummy airline
008 */
009@RemoteServiceRelativePath("airline")
010public interface AirlineService extends RemoteService {
011
012  /**
013   * Returns the current date and time on the server
014   */
015  Airline getAirline();
016
017  /**
018   * Always throws an undeclared exception so that we can see GWT handles it.
019   */
020  void throwUndeclaredException();
021
022  /**
023   * Always throws a declared exception so that we can see GWT handles it.
024   */
025  void throwDeclaredException() throws IllegalStateException;
026
027}
