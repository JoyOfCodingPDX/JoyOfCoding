001package edu.pdx.cs410J.grader;
002
003import org.junit.Test;
004
005import java.io.File;
006
007import static org.hamcrest.MatcherAssert.assertThat;
008import static org.hamcrest.core.IsEqual.equalTo;
009
010public class SubmitTest {
011
012  @Test
013  public void nineDigitStringIsInvalidLoginId() {
014    Submit submit = new Submit();
015    assertThat(submit.isNineDigitStudentId("123456789"), equalTo(true));
016  }
017
018  @Test
019  public void eightDigitStringIsValidLoginId() {
020    Submit submit = new Submit();
021    assertThat(submit.isNineDigitStudentId("12345678"), equalTo(false));
022  }
023
024  @Test
025  public void userNameIsValidLoginId() {
026    Submit submit = new Submit();
027    assertThat(submit.isNineDigitStudentId("whitlock"), equalTo(false));
028  }
029
030  @Test
031  public void emailAddressIsInvalidLoginId() {
032    Submit submit = new Submit();
033    assertThat(submit.looksLikeAnEmailAddress("me@email.com"), equalTo(true));
034  }
035
036  @Test
037  public void canSubmitClassesInStudentPackage() {
038    String userId = "student";
039
040    File file = makeFileWithPath("edu", "pdx", "cs410J", userId, "ProjectClass.java");
041    Submit submit = new Submit();
042    submit.setUserId(userId);
043    assertThat(submit.isInEduPdxCs410JDirectory(file), equalTo(true));
044  }
045
046  private File makeFileWithPath(String... path) {
047    File dir = new File(System.getProperty("user.dir"));
048
049    for (String part : path) {
050      dir = new File(dir, part);
051    }
052
053    return dir;
054  }
055
056  @Test
057  public void canSubmitClassesInSubPackage() {
058    String userId = "student";
059
060    File file = makeFileWithPath("edu", "pdx", "cs410J", userId, "subpackage", "ProjectClass.java");
061    Submit submit = new Submit();
062    submit.setUserId(userId);
063    assertThat(submit.isInEduPdxCs410JDirectory(file), equalTo(true));
064  }
065}
