001package edu.pdx.cs410J.grader;
002
003import org.hamcrest.Description;
004import org.hamcrest.TypeSafeMatcher;
005
006import java.util.Iterator;
007import java.util.List;
008import java.util.Spliterator;
009import java.util.function.Consumer;
010import java.util.function.Function;
011import java.util.stream.Stream;
012
013public class D2LStudentPropertyMatcher extends TypeSafeMatcher<List<GradesFromD2L.D2LStudent>> {
014
015  private final String expectedValue;
016  private final Function<GradesFromD2L.D2LStudent, String> accessor;
017  private final String propertyName;
018
019  public D2LStudentPropertyMatcher(Function<GradesFromD2L.D2LStudent, String> accessor, String propertyName, String expectedValue) {
020    this.expectedValue = expectedValue;
021    this.accessor = accessor;
022    this.propertyName = propertyName;
023  }
024
025  @Override
026  protected boolean matchesSafely(List<GradesFromD2L.D2LStudent> students) {
027    for (GradesFromD2L.D2LStudent student : students) {
028      if (this.accessor.apply(student).equals(expectedValue)) {
029        return true;
030      }
031    }
032
033    return false;
034  }
035
036  @Override
037  public void describeTo(Description description) {
038    description.appendText("a student with a " + propertyName + " of ").appendValue(expectedValue);
039  }
040
041  @Override
042  protected void describeMismatchSafely(List<GradesFromD2L.D2LStudent> students, Description mismatchDescription) {
043    mismatchDescription.appendText("the students' " + propertyName + "s were: ");
044    Stream<String> actualValues = students.stream().map(accessor);
045    mismatchDescription.appendValueList("", ",", "", toIterable(actualValues));
046  }
047
048  private <T> Iterable<T> toIterable(final Stream<T> stream) {
049    return new Iterable<T>() {
050      @Override
051      public Iterator<T> iterator() {
052        return stream.iterator();
053      }
054
055      @Override
056      public void forEach(Consumer<? super T> action) {
057        stream.forEach(action);
058      }
059
060      @Override
061      public Spliterator<T> spliterator() {
062        return stream.spliterator();
063      }
064    };
065  }
066}
