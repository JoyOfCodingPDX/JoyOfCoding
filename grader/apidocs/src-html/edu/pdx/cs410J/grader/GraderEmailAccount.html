041  private void fetchAttachmentsFromUnreadMessagesInFolder(Folder folder, EmailAttachmentProcessor processor) {
042    try {
043      Message[] messages = folder.getMessages();
044
045      FetchProfile profile = new FetchProfile();
046      profile.add(FetchProfile.Item.ENVELOPE);
047      profile.add(FetchProfile.Item.FLAGS);
048
049      folder.fetch(messages, profile);
050
051      for (Message message : messages) {
052        if (isUnread(message)) {
053          fetchAttachmentsFromUnreadMessage(message, processor);
054        }
055      }
056
057    } catch (MessagingException | IOException ex ) {
058      throw new IllegalStateException("While printing unread messages", ex);
059    }
060  }
061
062  @VisibleForTesting
063  protected void fetchAttachmentsFromUnreadMessage(Message message, EmailAttachmentProcessor processor) throws MessagingException, IOException {
064    printMessageInformation(message);
065    if (isMultipartMessage(message)) {
066      processAttachments(message, processor);
067    } else {
068      warnOfUnexpectedMessage(message, "Fetched a message that wasn't multipart: " + message.getContentType());
069    }
070  }
071
072  private void processAttachments(Message message, EmailAttachmentProcessor processor) throws MessagingException, IOException {
073    Multipart parts;
074    try {
075      parts = (Multipart) message.getContent();
076    } catch (IOException ex) {
077      throw new MessagingException("While getting content", ex);
078    }
079
080    debug("  Part count: " + parts.getCount());
081
082    if (parts.getCount() <= 0) {
083      warnOfUnexpectedMessage(message, "Fetched a message that has no attachments");
084    }
085
086    for (String contentType : processor.getSupportedContentTypes()) {
087      for (BodyPart part : getBodyPartsInReverseOrder(parts)) {
088        if (attemptToProcessPart(message, part, processor, contentType)) {
089          return;
090        }
091      }
092    }
093
094    warnOfUnexpectedMessage(message, "Could not process of any attachments");
095  }
096
097  private boolean attemptToProcessPart(Message message, BodyPart part, EmailAttachmentProcessor processor, String supportedContentType) throws MessagingException, IOException {
098    if (partHasContentType(part, supportedContentType)) {
099      debug("    Processing attachment of type " + part.getContentType());
100      processAttachmentFromPart(message, part, processor);
101      return true;
102
103    } else if (partIsMultiPart(part)) {
104      debug("    Attempting to process attachment of type " + part.getContentType());
105
106      for (BodyPart subpart : getBodyPartsInReverseOrder((Multipart) part.getContent())) {
107        if (attemptToProcessPart(message, subpart, processor, supportedContentType)) {
108          return true;
109        }
110      }
111
112    } else {
113      debug("    Skipping attachment of type " + part.getContentType());
114    }
115    return false;
116  }
117
118  private List<BodyPart> getBodyPartsInReverseOrder(Multipart parts) throws MessagingException {
119    List<BodyPart> list = new ArrayList<>(parts.getCount());
120    for (int i = parts.getCount() - 1; i >= 0; i--) {
121      list.add(parts.getBodyPart(i));
122    }
123
124    return list;
125  }
126
127  private boolean partIsMultiPart(BodyPart part) throws IOException, MessagingException {
128    return part.getContent() instanceof Multipart;
129  }
130
131  private boolean partHasContentType(BodyPart part, String contentType) throws MessagingException {
132    return part.getContentType().toUpperCase().contains(contentType.toUpperCase());
133  }
134
135  private void processAttachmentFromPart(Message message, BodyPart part, EmailAttachmentProcessor processor) throws MessagingException, IOException {
136    String fileName = part.getFileName();
137    processor.processAttachment(message, fileName, part.getInputStream());
138  }
139
140  private void warnOfUnexpectedMessage(Message message, String description) throws MessagingException {
141    debug(description);
142    printMessageDetails(message);
143  }
144
145  private boolean isMultipartMessage(Message message) throws MessagingException {
146    return message.isMimeType("multipart/*");
147  }
148
149  private boolean isUnread(Message message) throws MessagingException {
150    return !message.getFlags().contains(Flags.Flag.SEEN);
151  }
152
153  private void printMessageInformation(Message message) throws MessagingException {
154    debug("Message");
155    printMessageDetails(message);
156  }
157
158  private void printMessageDetails(Message message) throws MessagingException {
159    debug("  To: " + addresses(message.getRecipients(Message.RecipientType.TO)));
160    debug("  From: " + addresses(message.getFrom()));
161    debug("  Subject: " + message.getSubject());
162    debug("  Sent: " + message.getSentDate());
163    debug("  Flags: " + flags(message.getFlags()));
164    debug("  Content Type: " + message.getContentType());
165  }
166
167  private StringBuilder flags(Flags flags) {
168    StringBuilder sb = new StringBuilder();
169    systemFlags(flags, sb);
170    return sb;
171  }
172
173  private void systemFlags(Flags flags, StringBuilder sb) {
174    Flags.Flag[] systemFlags = flags.getSystemFlags();
175    for (int i = 0; i < systemFlags.length; i++) {
176      Flags.Flag flag = systemFlags[i];
177      if (flag == Flags.Flag.ANSWERED) {
178        sb.append("ANSWERED");
179
180      } else if (flag == Flags.Flag.DELETED) {
181        sb.append("DELETED");
182
183      } else if (flag == Flags.Flag.DRAFT) {
184        sb.append("DRAFT");
185
186      } else if (flag == Flags.Flag.FLAGGED) {
187        sb.append("FLAGGED");
188
189      } else if (flag == Flags.Flag.RECENT) {
190        sb.append("RECENT");
191
192      } else if (flag == Flags.Flag.SEEN) {
193        sb.append("SEEN");
194
195      } else if (flag == Flags.Flag.USER) {
196        sb.append("USER");
197
198      } else {
199        sb.append("UNKNOWN");
200      }
201
202      if (i > systemFlags.length - 1) {
203        sb.append(", ");
204      }
205    }
206  }
207
208  private String addresses(Address[] addresses) {
209    if (addresses == null) {
210      return "<None>";
211    }
212
213    StringBuilder sb = new StringBuilder();
214    for (int i = 0; i < addresses.length; i++) {
215      Address address = addresses[i];
216      sb.append(address.toString());
217      if (i > addresses.length - 1) {
218        sb.append(", ");
219      }
220    }
221
222    return sb.toString();
223  }

225  private void printFolderInformation(Folder folder) {
226    try {
227      debug("Folder: " + folder.getFullName());
228      debug("Message count: " + folder.getMessageCount());
229      debug("Unread messages: " + folder.getUnreadMessageCount());
230      debug("New messages: " + folder.getNewMessageCount());
231
232    } catch (MessagingException ex) {
233      throw new IllegalStateException("While getting folder information", ex);
234    }
235  }

241  private Folder openFolder(Store store, String folderName) {
242    try {
243      Folder folder = store.getDefaultFolder();
244      checkForValidFolder(folder, "Default");
245      folder = folder.getFolder(folderName);
246      folder.open(Folder.READ_WRITE);
247      return folder;
248
249    } catch (MessagingException ex) {
250      throw new IllegalStateException("While opening project submissions folder", ex);
251    }
252  }
253
254  private void checkForValidFolder(Folder folder, String folderName) {
255    if (folder == null) {
256      throw new IllegalStateException("Folder \"" + folderName + "\" does not exist");
257    }
258
259  }

261  private Store connectToIMAPServer() {
262    try {
263      Properties props = new Properties();
264
265      if (this.trustLocalhostSSL) {
266        MailSSLSocketFactory socketFactory= new MailSSLSocketFactory();
267        socketFactory.setTrustedHosts(new String[] { "127.0.0.1", "localhost" });
268        props.put("mail.imaps.ssl.socketFactory", socketFactory);
269      }
270
271      Session session = Session.getInstance(props, null);
272      Store store = session.getStore("imaps");
273      store.connect(this.emailServerHostName, this.emailServerPort, this.userName, this.password);
274      return store;
275
276    } catch (MessagingException | GeneralSecurityException ex) {
277      throw new IllegalStateException("While connecting to " + this.emailServerHostName + ":" + this.emailServerPort, ex);
278    }
279  }
280
281  public void fetchAttachmentsFromUnreadMessagesInFolder(String folderName, EmailAttachmentProcessor processor) {
282    Store store = connectToIMAPServer();
283    Folder folder = openFolder(store, folderName);
284    printFolderInformation(folder);
285
286    fetchAttachmentsFromUnreadMessagesInFolder(folder, processor);
287
288    try {
289      folder.close(false);
290      store.close();
291
292    } catch (MessagingException ex) {
293      throw new IllegalStateException("While closing folder and store", ex);
294    }
295
296  }
