038 * http://java.sun.com/products/javamail</a>

194  /**
195   * Validates the state of this submission
196   *
197   * @throws IllegalStateException If any state is incorrect or missing
198   */
199  private void validate() {
200    validateProjectName();
201
202    if (projName == null) {
203      throw new IllegalStateException("Missing project name");
204    }
205
206    if (userName == null) {
207      throw new IllegalStateException("Missing student name");
208    }
209
210    if (userId == null) {
211      throw new IllegalStateException("Missing login id");
212    }
213
214    if (isNineDigitStudentId(userId)) {
215      throw new IllegalStateException(loginIdShouldNotBeStudentId(userId));
216    }
217
218    if (looksLikeAnEmailAddress(userId)) {
219      throw new IllegalStateException(loginIdShouldNotBeEmailAddress(userId));
220    }
221
222    if (userEmail == null) {
223      throw new IllegalStateException("Missing email address");
224
225    } else {
226      // Make sure user's email is okay
227      try {
228        new InternetAddress(userEmail);
229
230      } catch (AddressException ex) {
231        String m = "Invalid email address: " + userEmail;
232        throw new IllegalStateException(m, ex);
233      }
234    }
235  }

239  }

267  /**
268   * Submits the project to the grader
269   *
270   * @param verify Should the user be prompted to verify his submission?
271   * @return Whether or not the submission actually occurred
272   * @throws IllegalStateException If no source files were found
273   */
274  public boolean submit(boolean verify) throws IOException, MessagingException {
275    // Recursively search the source directory for .java files
276    Set<File> sourceFiles = searchForSourceFiles(fileNames);
277
278    db(sourceFiles.size() + " source files found");
279
280    if (sourceFiles.size() == 0) {
281      String s = "No source files were found.";
282      throw new IllegalStateException(s);
283    } else if (sourceFiles.size() < 3) {
284      String s = "Too few source files were submitted. Each project requires at least 3 files be" + 
285                 " submitted. You only submitted " + sourceFiles.size() + " files";
286      throw new IllegalStateException(s);
287    }
288
289    // Verify submission with user
290    if (verify && !verifySubmission(sourceFiles)) {
291      // User does not want to submit
292      return false;
293    }
294
295    // Timestamp
296    this.submitTime = LocalDateTime.now();
297
298    // Create a temporary zip file to hold the source files
299    File zipFile = makeZipFileWith(sourceFiles);
300
301    if (this.sendEmails) {
302      // Send the zip file as an email attachment to the TA
303      mailTA(zipFile, sourceFiles);
304
305      if (sendReceipt) {
306        mailReceipt(sourceFiles);
307      }
308    }
309
310    return true;
311  }

322  /**
323   * Searches for the files given on the command line.  Ignores files
324   * that do not end in .java, or that appear on the "no submit" list.
325   * Files must reside in a directory named
326   * edu/pdx/cs410J/<studentId>.
327   */
328  private Set<File> searchForSourceFiles(Set<String> fileNames) {
329    // Files should be sorted by name
330    SortedSet<File> files =
331      new TreeSet<>(Comparator.comparing(File::toString));
332    populateWithFilesFromSubdirectories(files, fileNames);
333
334    files.removeIf((file) -> !canBeSubmitted(file));
335
336    return files;
337  }
338
339  private void populateWithFilesFromSubdirectories(SortedSet<File> allFiles, Set<String> fileNames) {
340    populateWithFilesFromSubdirectories(allFiles, fileNames.stream().map(name -> new File(name).getAbsoluteFile()));
341  }
342
343  private void populateWithFilesFromSubdirectories(SortedSet<File> allFiles, Stream<File> files) {
344    files.forEach((file) -> {
345      if (file.isDirectory()) {
346        populateWithFilesFromSubdirectories(allFiles, Arrays.stream(file.listFiles()));
347
348      } else {
349        allFiles.add(file);
350      }
351    });
352
353  }

355  protected boolean canBeSubmitted(File file) {
356    if (!fileExists(file)) {
357      return false;
358    }
359
360    // Is the file on the "no submit" list?
361    List<String> noSubmit = fetchListOfFilesThatCanNotBeSubmitted();
362    String name = file.getName();
363    if (noSubmit.contains(name)) {
364      err.println("** Not submitting file " + file +
365        " because it is on the \"no submit\" list");
366      return false;
367    }
368
369    // Verify that file is in the correct directory.
370    if (!isInAKoansDirectory(file) && !isInMavenProjectDirectory(file)) {
371      err.println("** Not submitting file " + file +
372        ": it does not reside in a Maven project in a directory named " +
373        "edu" + File.separator + "pdx" + File.separator +
374        "cs410J" + File.separator + userId + " (or in one of the koans directories)");
375      return false;
376    }
377
378    // Does the file name end in .java?
379    if (!canFileBeSubmitted(name)) {
380      err.println("** Not submitting file " + file +
381        " because does end in \".java\"");
382      return false;
383    }
384
385    return true;
386  }

416  private boolean isInAKoansDirectory(File file) {
417    boolean isInAKoansDirectory = hasParentDirectories(file, "beginner") ||
418      hasParentDirectories(file, "intermediate") ||
419      hasParentDirectories(file, "advanced") ||
420      hasParentDirectories(file, "java7") ||
421      hasParentDirectories(file, "java8");
422    if (isInAKoansDirectory) {
423      this.isSubmittingKoans = true;
424      db(file + " is in a koans directory");
425    }
426    return isInAKoansDirectory;
427  }

429  @VisibleForTesting
430  boolean isInMavenProjectDirectory(File file) {
431    boolean isInMavenProjectDirectory = hasParentDirectories(file, userId, "cs410J", "pdx", "edu", "java|javadoc|resources", "main|test|it", "src");
432    if (isInMavenProjectDirectory) {
433      db(file + " is in the edu/pdx/cs410J directory");
434    }
435    return isInMavenProjectDirectory;

438  private boolean hasParentDirectories(File file, String... parentDirectoryNames) {
439    File parent = file.getParentFile();
440
441    // Skip over subpackages
442    while (parent != null && !parent.getName().equals(parentDirectoryNames[0])) {
443      parent = parent.getParentFile();
444    }
445
446    for (String parentDirectoryName : parentDirectoryNames) {
447      if (parent == null || !parent.getName().matches(parentDirectoryName)) {
448        return false;
449
450      } else {
451        parent = parent.getParentFile();
452      }
453    }
454
455    return true;
456  }

462  private List<String> fetchListOfStringsFromUrl(String listUrl) {
463    if (!failIfDisallowedFiles) {
464      return Collections.emptyList();
465    }
466
467    List<String> strings = new ArrayList<>();
468
469    try {
470      URL url = new URL(listUrl);
471      InputStreamReader isr = new InputStreamReader(url.openStream());
472      BufferedReader br = new BufferedReader(isr);
473      while (br.ready()) {
474        strings.add(br.readLine().trim());
475      }
476
477    } catch (MalformedURLException ex) {
478      err.println("** WARNING: Cannot access " + listUrl + ": " +
479        ex.getMessage());
480
481    } catch (IOException ex) {
482      err.println("** WARNING: Problems while reading " + listUrl + ": " + ex.getMessage());
483    }
484    return strings;
485  }

487  /**
488   * Prints a summary of what is about to be submitted and prompts the
489   * user to verify that it is correct.
490   *
491   * @return <code>true</code> if the user wants to submit
492   */
493  private boolean verifySubmission(Set<File> sourceFiles) {
494    // Print out what is going to be submitted
495    out.print("\n" + userName);
496    out.print("'s submission for ");
497    out.println(projName);
498
499    for (File file : sourceFiles) {
500      out.println("  " + file);
501    }
502
503    if (comment != null) {
504      out.println("\nComment: " + comment + "\n\n");
505    }
506
507    out.println("A receipt will be sent to: " + userEmail + "\n");
508
509    warnIfMainProjectClassIsNotSubmitted(sourceFiles);
510
511    warnIfTestClassesAreNotSubmitted(sourceFiles);
512
513    return doesUserWantToSubmit();
514  }

516  protected void warnIfMainProjectClassIsNotSubmitted(Set<File> sourceFiles) {
517    boolean wasMainProjectClassSubmitted = sourceFiles.stream().anyMatch((f) -> f.getName().contains(this.projName));
518    if (!wasMainProjectClassSubmitted && !this.isSubmittingKoans) {
519      String mainProjectClassName = this.projName + ".java";
520      out.println("*** WARNING: You are submitting " + this.projName +
521        ", but did not include " + mainProjectClassName + ".\n" +
522        "    You might want to check the name of the project or the files you are submitting.\n");
523    }
524  }
525
526  protected void warnIfTestClassesAreNotSubmitted(Set<File> sourceFiles) {
527    boolean wereTestClassessSubmitted = sourceFiles.stream().anyMatch((f) -> f.getName().contains("test"));
528    if (!wereTestClassessSubmitted && !this.isSubmittingKoans) {
529      out.println("*** WARNING: You are not submitting a \"test\" directory.\n" +
530        "    Your unit tests are executed as part of the grading of your project.\n");
531    }
532  }

534  private boolean doesUserWantToSubmit() {
535    InputStreamReader isr = new InputStreamReader(System.in);
536    BufferedReader in = new BufferedReader(isr);
537
538    while (true) {
539      out.print("Do you wish to continue with the submission? (yes/no) ");
540      out.flush();
541
542      try {
543        String line = in.readLine().trim();
544        switch (line) {
545          case "yes":
546            return true;
547
548          case "no":
549            return false;
550
551          default:
552            err.println("** Please enter yes or no");
553            break;
554        }
555
556      } catch (IOException ex) {
557        err.println("** Exception while reading from System.in: " + ex);
558      }
559    }
560  }

562  /**
563   * Returns the name of a <code>File</code> relative to the source
564   * directory.

566  protected String getZipEntryNameFor(File file) {
567    if (isSubmittingKoans) {
568      return file.getParentFile().getName() + "/" + file.getName();
569
570    } else {
571      // We already know that the file is in the correct directory
572      return getZipEntryNameFor(file.getPath());
573    }

576  @VisibleForTesting
577  static String getZipEntryNameFor(String filePath) {
578    Pattern pattern = Pattern.compile(".*/src/(main|test|it)/(java|javadoc|resources)/edu/pdx/cs410J/(.*)");
579    Matcher matcher = pattern.matcher(filePath);
580
581    if (matcher.matches()) {
582      return "src/" + matcher.group(1) + "/" + matcher.group(2) + "/edu/pdx/cs410J/" + matcher.group(3);
583    } else {
584      throw new IllegalStateException("Can't extract zip entry name for " + filePath);
585    }
586  }

588  /**
589   * Creates a Zip file that contains the source files.  The Zip File
590   * is temporary and is deleted when the program exits.
591   */
592  private File makeZipFileWith(Set<File> sourceFiles) throws IOException {
593    String zipFileName = userName.replace(' ', '_') + "-TEMP";
594    File zipFile = File.createTempFile(zipFileName, ".zip");
595    if (!saveZip) {
596      zipFile.deleteOnExit();
597
598    } else {
599      out.println("Saving temporary Zip file: " + zipFile);
600    }
601
602    db("Created Zip file: " + zipFile);
603
604    Map<File, String> sourceFilesWithNames =
605      sourceFiles.stream().collect(Collectors.toMap(file -> file, this::getZipEntryNameFor));
606
607    new ZipFileOfFilesMaker(sourceFilesWithNames, zipFile, getManifestEntries()).makeZipFile();
608    return zipFile;
609  }
610
611  private Map<Attributes.Name, String> getManifestEntries() {
612    Map<Attributes.Name, String> manifestEntries = new HashMap<>();
613    manifestEntries.put(ManifestAttributes.USER_NAME, userName);
614    manifestEntries.put(ManifestAttributes.USER_ID, userId);
615    manifestEntries.put(ManifestAttributes.USER_EMAIL, userEmail);
616    manifestEntries.put(ManifestAttributes.PROJECT_NAME, projName);
617    manifestEntries.put(ManifestAttributes.SUBMISSION_COMMENT, comment);
618    manifestEntries.put(ManifestAttributes.SUBMISSION_TIME, ManifestAttributes.formatSubmissionTime(submitTime));
619    return manifestEntries;
620  }

622  /**
623   * Sends the Zip file to the TA as a MIME attachment.  Also includes
624   * a textual summary of the contents of the Zip file.
625   */
626  private void mailTA(File zipFile, Set<File> sourceFiles) throws MessagingException {
627    MimeMessage message =
628      newEmailTo(newEmailSession(debug), TA_EMAIL)
629        .from(userEmail, userName)
630        .withSubject("CS410J-SUBMIT " + userName + "'s " + projName)
631        .createMessage();
632
633    MimeBodyPart textPart = createTextPartOfTAEmail(sourceFiles);
634    MimeBodyPart filePart = createZipAttachment(zipFile);
635
636    Multipart mp = new MimeMultipart();
637    mp.addBodyPart(textPart);
638    mp.addBodyPart(filePart);
639
640    message.setContent(mp);
641
642    out.println("Submitting project to Grader");
643
644    Transport.send(message);
645  }
646
647  private MimeBodyPart createZipAttachment(File zipFile) throws MessagingException {
648    // Now attach the Zip file
649    DataSource ds = new FileDataSource(zipFile) {
650      @Override
651      public String getContentType() {
652        return "application/zip";
653      }
654    };
655    DataHandler dh = new DataHandler(ds);
656    MimeBodyPart filePart = new MimeBodyPart();
657
658    String zipFileTitle = userName.replace(' ', '_') + ".zip";
659
660    filePart.setDataHandler(dh);
661    filePart.setFileName(zipFileTitle);
662    filePart.setDescription(userName + "'s " + projName);
663
664    return filePart;
665  }

667  private MimeBodyPart createTextPartOfTAEmail(Set<File> sourceFiles) throws MessagingException {
668    // Create the text portion of the message
669    StringBuilder text = new StringBuilder();
670    text.append("Student name: ").append(userName).append(" (").append(userEmail).append(")\n");
671    text.append("Project name: ").append(projName).append("\n");
672    text.append("Submitted on: ").append(humanReadableSubmitDate()).append("\n");
673    if (comment != null) {
674      text.append("\nComment: ").append(comment).append("\n\n");
675    }
676    text.append("Contents:\n");
677
678    for (File file : sourceFiles) {
679      text.append("  ").append(getZipEntryNameFor(file)).append("\n");
680    }
681    text.append("\n\n");
682
683    MimeBodyPart textPart = new MimeBodyPart();
684    textPart.setContent(text.toString(), "text/plain");
685
686    // Try not to display text as separate attachment
687    textPart.setDisposition("inline");
688    return textPart;
689  }

695  /**
696   * Sends a email to the user as a receipt of the submission.
697   */
698  private void mailReceipt(Set<File> sourceFiles) throws MessagingException {
699    String subject = "CS410J " + projName + " submission";
700    InternetAddress email = newInternetAddress(this.userEmail, this.userName);
701    MimeMessage message =
702      newEmailTo(newEmailSession(debug), email)
703        .from(TA_EMAIL)
704        .replyTo(DAVE_EMAIL)
705        .withSubject(subject)
706        .createMessage();
707
708    // Create the contents of the message
709    StringBuilder text = new StringBuilder();
710    text.append("On ").append(humanReadableSubmitDate()).append("\n");
711    text.append(userName).append(" (").append(userEmail).append(")\n");
712    text.append("submitted the following files for ").append(projName).append(":\n");
713
714    for (File file : sourceFiles) {
715      text.append("  ").append(file.getAbsolutePath()).append("\n");
716    }
717
718    if (comment != null) {
719      text.append("\nComment: ").append(comment).append("\n\n");
720    }
721
722    text.append("\n\n");
723    text.append("Have a nice day.");
724
725    // Add the text to the message and send it
726    message.setText(text.toString());
727    message.setDisposition("inline");
728
729    out.println("Sending receipt to you at " + userEmail);
730
731    Transport.send(message);
732  }

766  /**
767   * Parses the command line, finds the source files, prompts the user
768   * to verify whether or not the settings are correct, and then sends
769   * an email to the Grader.
770   */
771  void parseCommandLineAndSubmit(String[] args) throws IOException, MessagingException {
772    // Parse the command line
773    for (int i = 0; i < args.length; i++) {
774      // Check for options first
775      if (args[i].equals("-smtp")) {
776        if (++i >= args.length) {
777          usage("No SMTP server specified");
778        }
779
780        this.setEmailServerHostName(args[i]);
781
782      } else if (args[i].equals("-verbose")) {
783        this.setDebug(true);
784
785      } else if (args[i].equals("-savezip")) {
786        this.setSaveZip(true);
787
788      } else if (args[i].equals("-nosend")) {
789        this.setSendEmails(false);
790
791      } else if (args[i].equals("-comment")) {
792        if (++i >= args.length) {
793          usage("No comment specified");
794        }
795
796        this.setComment(args[i]);
797
798      } else if (this.projName == null) {
799        this.setProjectName(args[i]);
800
801      } else if (this.userName == null) {
802        this.setUserName(args[i]);
803
804      } else if (this.userId == null) {
805        this.setUserId(args[i]);
806
807      } else if (this.userEmail == null) {
808        this.setUserEmail(args[i]);
809
810      } else {
811        // The name of a source file
812        this.addFile(args[i]);
813      }
814    }
815
816    boolean submitted;
817
818    try {
819      // Make sure that user entered enough information
820      this.validate();
821
822      this.db("Command line successfully parsed.");
823
824      submitted = this.submit(true);
825
826    } catch (IllegalStateException ex) {
827      usage(ex.getMessage());
828      return;
829    }
830
831    // All done.
832    if (submitted) {
833      out.println(this.projName + " submitted successfully.  Thank you.");
834
835    } else {
836      out.println(this.projName + " not submitted.");
837    }
838  }
