030  /**
031   * Returns a textual summary of a <code>Student</code>
032   */
033  private static String getSummary(Student student) {
034    StringBuilder sb = new StringBuilder();
035    sb.append("Name: ").append(student.getFullName()).append("\n");
036    sb.append("UNIX login: ").append(student.getId()).append("\n");
037    if (student.getEmail() != null) {
038      sb.append("Email: ").append(student.getEmail()).append("\n");
039    }
040    if (student.getSsn() != null) {
041      sb.append("Student id: ").append(student.getSsn()).append("\n");
042    }
043    if (student.getMajor() != null) {
044      sb.append("Major: ").append(student.getMajor()).append("\n");
045    }
046    sb.append("Enrolled in: ").append(student.getEnrolledSection().asString()).append("\n");
047    return sb.toString();
048  }

050  /**
051   * Ask the student a question and return his response
052   */
053  private static String ask(String question) {
054    out.print(breakUpInto80CharacterLines(question));
055    out.print(" ");
056    out.flush();
057
058    String response = null;
059    try {
060      response = in.readLine();
061
062    } catch (IOException ex) {
063      printErrorMessageAndExit("** IOException while reading response", ex);
064    }
065
066    return response;
067  }

081  public static void main(String[] args) {
082    parseCommandLine(args);

096  private static void addNotesToStudent(Student student, String learn, String comments) {
097    if (isNotEmpty(learn)) {
098      student.addNote(student.getFullName() + " would like to learn " + learn);
099    }
100
101    if (isNotEmpty(comments)) {
102      student.addNote(student.getFullName() + " has these comments: " + comments);
103    }
104  }

106  private static Student gatherStudentInformation() {
107    String firstName = ask("What is your first name?");
108    String lastName = ask("What is your last name?");
109    String nickName = ask("What is your nickname? (Leave blank if " +
110                          "you don't have one)");
111    String id = ask("MANDATORY: What is your UNIX login id?");
112
113    if (isEmpty(id)) {
114      printErrorMessageAndExit("** You must enter a valid UNIX login id");
115
116    } else if (isEmailAddress(id)) {
117      printErrorMessageAndExit("** Your student id cannot be an email address");
118    }
119
120    Student student = new Student(id);
121    setValueIfNotEmpty(firstName, student::setFirstName);
122    setValueIfNotEmpty(lastName, student::setLastName);
123    setValueIfNotEmpty(nickName, student::setNickName);
124
125    askQuestionAndSetValue("What is your email address (doesn't have to be PSU)?", student::setEmail);
126    askQuestionAndSetValue("What is your student id (XXXXXXXXX)?", student::setSsn);
127    askQuestionAndSetValue("What is your major?", student::setMajor);
128
129    askEnrolledSectionQuestion(student);
130
131    return student;
132  }

134  @VisibleForTesting
135  static boolean isEmailAddress(String id) {
136    try {
137      boolean strict = true;
138      new InternetAddress(id, strict);
139      return true;
140
141    } catch (AddressException e) {
142      return false;
143    }
144  }

146  private static void askEnrolledSectionQuestion(Student student) {
147    String answer = ask("MANDATORY: Are you enrolled in the undergraduate or graduate section of this course? [u/g]");
148    if (isEmpty(answer)) {
149      printErrorMessageAndExit("Missing enrolled section. Please enter a \"u\" or \"g\"");
150    }
151
152    if (answer.toLowerCase().startsWith("u")) {
153      student.setEnrolledSection(Student.Section.UNDERGRADUATE);
154
155    } else if (answer.toLowerCase().startsWith("g")) {
156      student.setEnrolledSection(Student.Section.GRADUATE);
157
158    } else {
159      printErrorMessageAndExit("Unknown section \"" + answer + "\".  Please enter a \"u\" or \"g\"");
160    }
161
162  }

164  private static void askQuestionAndSetValue(String question, Consumer<String> setter) {
165    String answer = ask(question);
166    setValueIfNotEmpty(answer, setter);
167  }
168
169  static void setValueIfNotEmpty(String string, Consumer<String> setter) {
170    if (isNotEmpty(string)) {
171      setter.accept(string);
172    }
173  }

175  private static void emailSurveyResults(Student student, String learn, String comments) {
176    String summary = verifyInformation(student);
177
178    // Email the results of the survey to the TA and CC the student
179    out.println("Emailing your information to the Grader");
180
181    MimeMessage message = createEmailMessage(student);
182    MimeBodyPart textPart = createEmailText(learn, comments, summary);
183    MimeBodyPart xmlFilePart = createXmlAttachment(student);
184    addAttachmentsAndSendEmail(message, textPart, xmlFilePart);
185  }
186
187  private static void addAttachmentsAndSendEmail(MimeMessage message, MimeBodyPart textPart, MimeBodyPart filePart) {
188    // Finally, add the attachments to the message and send it
189    try {
190      Multipart mp = new MimeMultipart();
191      mp.addBodyPart(textPart);
192      mp.addBodyPart(filePart);
193
194      message.setContent(mp);
195
196      Transport.send(message);
197
198      logSentEmail(message);
199
200    } catch (MessagingException ex) {
201      printErrorMessageAndExit("** Exception while adding parts and sending", ex);
202    }
203  }
204
205  private static void logSentEmail(MimeMessage message) throws MessagingException {
206    StringBuilder sb = new StringBuilder();
207    sb.append("\nAn email with with subject \"");
208    sb.append(message.getSubject());
209    sb.append("\" was sent to ");
210
211    Address[] recipients = message.getAllRecipients();
212    for (int i = 0; i < recipients.length; i++) {
213      Address recipient = recipients[i];
214      sb.append(recipient);
215      if (i < recipients.length - 2) {
216        sb.append(", ");
217
218      } else if (i < recipients.length - 1) {
219        sb.append(" and ");
220      }
221    }
222
223    out.println(breakUpInto80CharacterLines(sb.toString()));
224  }

226  private static MimeBodyPart createXmlAttachment(Student student) {
227    byte[] xmlBytes = getXmlBytes(student);
228
229    if (saveStudentXmlFile) {
230      writeStudentXmlToFile(xmlBytes, student);
231    }
232
233    DataSource ds = new ByteArrayDataSource(xmlBytes, "text/xml");
234    DataHandler dh = new DataHandler(ds);
235    MimeBodyPart filePart = new MimeBodyPart();
236    try {
237      String xmlFileTitle = student.getId() + ".xml";
238
239      filePart.setDataHandler(dh);
240      filePart.setFileName(xmlFileTitle);
241      filePart.setDescription("XML file for " + student.getFullName());
242
243    } catch (MessagingException ex) {
244      printErrorMessageAndExit("** Exception with file part", ex);
245    }
246    return filePart;
247  }

249  private static void writeStudentXmlToFile(byte[] xmlBytes, Student student) {
250    File directory = new File(System.getProperty("user.dir"));
251    File file = new File(directory, student.getId() + ".xml");
252    try (FileOutputStream fos = new FileOutputStream(file)) {
253      fos.write(xmlBytes);
254      fos.flush();
255
256    } catch (IOException e) {
257      printErrorMessageAndExit("Could not write student XML file: " + file, e);
258    }
259
260    out.println("\nSaved student XML file to " + file + "\n");
261  }

263  private static MimeBodyPart createEmailText(String learn, String comments, String summary) {
264    // Create the text portion of the message
265    StringBuilder text = new StringBuilder();
266    text.append("Results of CS410J Survey:\n\n");
267    text.append(summary);
268    text.append("\n\nWhat do you hope to learn in CS410J?\n\n");
269    text.append(learn);
270    text.append("\n\nIs there anything else you'd like to tell me?\n\n");
271    text.append(comments);
272    text.append("\n\nThanks for filling out this survey!\n\nDave");
273
274    MimeBodyPart textPart = new MimeBodyPart();
275    try {
276      textPart.setContent(text.toString(), "text/plain");
277
278      // Try not to display text as separate attachment
279      textPart.setDisposition("inline");
280
281    } catch (MessagingException ex) {
282      printErrorMessageAndExit("** Exception with text part", ex);
283    }
284    return textPart;
285  }

287  private static MimeMessage createEmailMessage(Student student) {
288    MimeMessage message = null;
289    try {
290      String studentEmail = student.getEmail();
291      message = newEmailTo(newEmailSession(false), studentEmail, TA_EMAIL, "CS410J Survey for " + student.getFullName());
292
293      if (studentEmail != null) {
294        InternetAddress[] cc = {new InternetAddress(studentEmail)};
295        message.setRecipients(Message.RecipientType.CC, cc);
296      }
297
298    } catch (AddressException ex) {
299      printErrorMessageAndExit("** Exception with email address", ex);
300
301    } catch (MessagingException ex) {
302      printErrorMessageAndExit("** Exception while setting recipients email", ex);
303    }
304    return message;
305  }

307  private static byte[] getXmlBytes(Student student) {
308    // Create a temporary "file" to hold the Student's XML file.  We
309    // use a byte array so that potentially sensitive data (SSN, etc.)
310    // is not written to disk
311    byte[] bytes = null;
312
313    Document xmlDoc = XmlDumper.toXml(student);
314
315
316    try {
317      bytes = XmlHelper.getBytesForXmlDocument(xmlDoc);
318
319    } catch (TransformerException ex) {
320      ex.printStackTrace(System.err);
321      System.exit(1);
322    }
323    return bytes;
324  }

326  private static String verifyInformation(Student student) {
327    String summary = getSummary(student);
328
329    out.println("\nYou entered the following information about " +
330                "yourself:\n");
331    out.println(summary);
332
333    String verify = ask("\nIs this information correct (y/n)?");
334    if (!verify.equals("y")) {
335      printErrorMessageAndExit("** Not sending information.  Exiting.");
336    }
337    return summary;
338  }

360  private static void printIntroduction() {
361    // Ask the student a bunch of questions
362    String welcome =
363      "Welcome to the CS410J Survey Program.  I'd like to ask you a couple of " +
364      "questions about yourself.  Except for your UNIX login id and the section " +
365      "that you are enrolled in, no question" +
366      "is mandatory.  Your answers will be emailed to the TA and a receipt " +
367      "will be emailed to you.";
368
369    out.println("");
370    out.println(breakUpInto80CharacterLines(welcome));
371    out.println("");
372  }

374  @VisibleForTesting
375  static String breakUpInto80CharacterLines(String message) {
376    StringBuilder sb = new StringBuilder();
377    int currentLineLength = 0;
378    String[] words = message.split(" ");
379    for (String word : words) {
380      if (currentLineLength + word.length() > 80) {
381        sb.append('\n');
382        sb.append(word);
383        currentLineLength = word.length();
384
385      } else {
386        if (currentLineLength > 0) {
387          sb.append(' ');
388        }
389        sb.append(word);
390        currentLineLength += word.length() + 1;
391      }
392
393    }
394    return sb.toString();
395  }

397  private static void parseCommandLine(String[] args) {
398    // Parse the command line
399    for (int i = 0; i < args.length; i++) {
400      String arg = args[i];
401      if (arg.equals("-mailServer")) {
402        if (++i >= args.length) {
403          err.println("** Missing mail server name");
404          usage();
405        }
406
407        serverName = arg;
408
409      } else if (arg.equals("-saveStudentXmlFile")) {
410        saveStudentXmlFile = true;
411
412      } else if (arg.startsWith("-")) {
413        err.println("** Unknown command line option: " + arg);
414        usage();
415
416      } else {
417        err.println("** Spurious command line: " + arg);
418        usage();
419      }
420    }
421  }
