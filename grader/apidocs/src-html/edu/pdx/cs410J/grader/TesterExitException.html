003/**
004 * This exception is thrown when the
005 * <code>TesterSecurityManager</code> attempts to exit the JVM.  This
006 * allows us to execute multiple <code>main</code> methods inside the
007 * same JVM.  Note that it must subclass
008 * <code>RuntimeException</code>.  If it didn't we'd have to
009 * explicitly declare it in the <code>throws</code> clause.
010 */
011@SuppressWarnings("serial")
012public class TesterExitException extends RuntimeException {
013
014  /**
015   * Create a new <code>TesterExitException</code>
016   */
017  public TesterExitException(String message) {
018    super(message);
019  }
020
021}
