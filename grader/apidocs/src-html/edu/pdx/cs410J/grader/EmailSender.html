001package edu.pdx.cs410J.grader;
002
003import com.google.common.annotations.VisibleForTesting;
004
005import javax.mail.Message;
006import javax.mail.MessagingException;
007import javax.mail.Session;
008import javax.mail.internet.InternetAddress;
009import javax.mail.internet.MimeMessage;
010import java.util.Properties;
011
012public class EmailSender {
013  /**
014   * The grader's email address
015   */
016  @VisibleForTesting
017  static final String TA_EMAIL = "sjavata@gmail.com";
018  /**
019   * The name of the SMTP server that is used to send email
020   */
021  protected static String serverName = "mailhost.cs.pdx.edu";
022
023  private static int emailServerPort = 25;
024
025  protected static MimeMessage newEmailTo(Session session, String recipient, String subject) throws MessagingException {
026    return newEmailTo(session, TA_EMAIL, recipient, subject);
027  }
028
029  protected static MimeMessage newEmailTo(Session session, String sender, String recipient, String subject) throws MessagingException {
030    // Make a new email message
031    MimeMessage message = new MimeMessage(session);
032
033    InternetAddress[] to = {new InternetAddress(recipient)};
034    message.setRecipients(Message.RecipientType.TO, to);
035    message.setFrom(sender);
036    message.setSubject(subject);
037    return message;
038  }
039
040  protected static Session newEmailSession(boolean debug) {
041    // Obtain a Session for sending email
042    Properties props = new Properties();
043    props.put("mail.smtp.host", serverName);
044    props.put("mail.smtp.port", emailServerPort);
045    props.put("mail.smtp.localhost", "127.0.0.1");
046    Session session = Session.getDefaultInstance(props, null);
047    session.setDebug(debug);
048    return session;
049  }
050
051  public void setEmailServerPort(int port) {
052    emailServerPort = port;
053  }
054}
