37    public void canSubmitClassesInStudentPackage() {
38      String userId = "student";
39  
40      File file = makeFileWithPath("edu", "pdx", "cs410J", userId, "ProjectClass.java");
41      Submit submit = new Submit();

43      assertThat(submit.isInEduPdxCs410JDirectory(file), equalTo(true));
44    }
45  
46    private File makeFileWithPath(String... path) {
47      File dir = new File(System.getProperty("user.dir"));
48  
49      for (String part : path) {
50        dir = new File(dir, part);
51      }
52  
53      return dir;
54    }
