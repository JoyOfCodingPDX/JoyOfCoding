55    @Test
56    public void assignmentSelectedFireWhenAssignmentSelectedInView() {
57      ArgumentCaptor<AssignmentSelectedHandler> viewHandler = ArgumentCaptor.forClass(AssignmentSelectedHandler.class);
58      verify(this.view).addAssignmentSelectedHandler(viewHandler.capture());
59  
60      AssignmentSelectedEventHandler eventHandler = mock(AssignmentSelectedEventHandler.class);
61      this.bus.register(eventHandler);
62  
63      this.bus.post(new GradeBookLoaded(book));
64  
65      viewHandler.getValue().assignmentSelected(1);
66  
67      ArgumentCaptor<AssignmentSelectedEvent> event = ArgumentCaptor.forClass(AssignmentSelectedEvent.class);
68      verify(eventHandler).handle(event.capture());
69  
70      assertThat(event.getValue().getAssignment(), equalTo(assignment1));
71    }
