37    @Test
38    public void matchStudentBasedOnFirstAndLastName() throws StudentEmailAttachmentProcessor.SubmissionException {
39      String projectName = "Project";
40  
41      GradeBook gradebook = createGradeBookWithAssignment(projectName);
42      Student student = createStudentInGradeBook(gradebook);
43  
44      String studentName = student.getFirstName() + " " + student.getLastName();
45      String wrongStudentId = "Not the student id we expect";
46      String wrongEmail = "Not the email that we expect";
47  
48      Manifest manifest = manifest(projectName)
49        .setStudentName(studentName)
50        .setStudentId(wrongStudentId)
51        .setStudentEmail(wrongEmail)
52        .build();
53  
54      noteProjectSubmissionInGradeBook(gradebook, manifest);
55  
56      assertThat(gradebook.getStudent(wrongStudentId), isNotPresent());
57  
58      assertThatProjectSubmissionWasRecordedForStudent(projectName, student);
59    }

168   @Test
169   public void submissionsPastDueDateAreLate() throws StudentEmailAttachmentProcessor.SubmissionException {
170     String projectName = "Project";
171     LocalDateTime dueDate = LocalDateTime.now();
172 
173     GradeBook gradebook = createGradeBookWithAssignment(projectName, dueDate);
174     Student student = createStudentInGradeBook(gradebook);
175 
176     LocalDateTime submissionDate = dueDate.plusDays(3);
177     Manifest manifest = manifest(projectName)
178       .setStudent(student)
179       .setSubmissionDate(submissionDate)
180       .build();
181 
182     noteProjectSubmissionInGradeBook(gradebook, manifest);
183 
184     assertThat(student.getLate(), contains(projectName));
185 
186     Grade grade = student.getGrade(projectName);
187     assertThat(grade, notNullValue());
188     List<Grade.SubmissionInfo> submissions = grade.getSubmissionInfos();
189     assertThat(submissions, hasSize(1));
190     Grade.SubmissionInfo submission = submissions.get(0);
191     assertThat(submission.getSubmissionTime(), equalTo(submissionDate));
192     assertThat(submission.isLate(), equalTo(true));
193   }

258   private static Matcher<? super Optional<Student>> isNotPresent() {
259     return new TypeSafeMatcher<>() {
260       @Override
261       protected boolean matchesSafely(Optional<Student> item) {
262         return item.isEmpty();
263       }
264 
265       @Override
266       public void describeTo(Description description) {
267         description.appendText("an Optional<Student> that is not present");
268       }
269     };
270   }

280   private static class ManifestBuilder {

282 
283     private ManifestBuilder setAttribute(Attributes.Name name, String value) {
284       this.manifest.getMainAttributes().put(name, value);
285       return this;
286     }
287 
288     Manifest build() {
289       return manifest;
290     }

292     private ManifestBuilder setProjectName(String projectName) {
293       return setAttribute(PROJECT_NAME, projectName);
