30    private static Path getMockPath(boolean exists) {
31      Path testOutput = mock(Path.class);
32  
33      FileSystemProvider provider = mock(FileSystemProvider.class);
34      when(provider.exists(testOutput)).thenReturn(exists);
35  
36      FileSystem fileSystem = mock(FileSystem.class);
37      when(fileSystem.provider()).thenReturn(provider);
38      when(testOutput.getFileSystem()).thenReturn(fileSystem);
39  
40      Path parent = mock(Path.class);
41      when(parent.getFileSystem()).thenReturn(fileSystem);
42      when(testOutput.getParent()).thenReturn(parent);
43      when(provider.exists(parent)).thenReturn(true);
44  
45      assertThat(Files.exists(testOutput), equalTo(exists));
46      return testOutput;
47    }

49    @Test
50    void submissionWithNoTestOutputNeedsToBeTested() {
51      FindUngradedSubmissions.SubmissionDetailsProvider submissionDetailsProvider = mock(FindUngradedSubmissions.SubmissionDetailsProvider.class);
52  
53      String studentId = "student123";
54      FindUngradedSubmissions.SubmissionDetails submissionDetails = new FindUngradedSubmissions.SubmissionDetails(studentId, LocalDateTime.now());
55      Path submission = getPathToExistingFile();
56      when(submissionDetailsProvider.getSubmissionDetails(submission)).thenReturn(submissionDetails);
57  
58      Path testOutput = getPathToNonExistingFile();
59  
60      FindUngradedSubmissions.TestOutputPathProvider testOutputProvider = mock(FindUngradedSubmissions.TestOutputPathProvider.class);
61      when(testOutputProvider.getTestOutput(any(Path.class), eq(studentId))).thenReturn(testOutput);
62  
63      FindUngradedSubmissions finder = new FindUngradedSubmissions(submissionDetailsProvider, testOutputProvider, mock(FindUngradedSubmissions.TestOutputDetailsProvider.class));
64      FindUngradedSubmissions.SubmissionAnalysis analysis = finder.analyzeSubmission(submission);
65      assertThat(analysis.needsToBeTested(), equalTo(true));
66      assertThat(analysis.needsToBeGraded(), equalTo(true));
67      assertThat(analysis.submission(), equalTo(submission));
68    }
