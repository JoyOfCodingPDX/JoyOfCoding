40    public static void main(String[] args) {
41      Thread.currentThread().setUncaughtExceptionHandler((t, e) -> printStackTrace(e));
42  
43      Injector injector = Guice.createInjector(new POAGraderUIModule());
44  
45      EventBus bus = injector.getInstance(EventBus.class);
46      registerEventLogger(bus);
47  
48      PlanOfAttackGrader ui = injector.getInstance(PlanOfAttackGrader.class);
49      ui.display();
50  
51      if (args.length >= 1) {
52        String gradeBookFileName = args[0];
53        File gradeBookFile = new File(gradeBookFileName);
54        bus.post(new LoadGradeBook(gradeBookFile));
55      }
56  
57      DownloadPOASubmissionsRequest downloadPOASubmissionsRequest;
58      if (args.length >= 3) {
59        String email = args[1];
60        String password = loadPasswordFromFile(args[2]);
61        downloadPOASubmissionsRequest = new DownloadPOASubmissionsRequest(email, password);
62  
63      } else {
64        downloadPOASubmissionsRequest = new DownloadPOASubmissionsRequest();
65      }
66  
67      bus.post(downloadPOASubmissionsRequest);
68    }

70    private static String loadPasswordFromFile(String fileName) {
71      File file = new File(fileName);
72      if (!file.exists()) {
73        System.err.println("Password file does not exist: " + file);
74        System.exit(1);
75      }
76  
77      try (BufferedReader br = new BufferedReader(new FileReader(file))) {
78        return br.readLine();
79  
80      } catch (IOException e) {
81        System.err.println("While reading password from " + file);
82        e.printStackTrace(System.err);
83        System.exit(1);
84        return null;
85      }
86    }

88    private static void printStackTrace(Throwable e) {

93    private void display() {
94      Dimension fullScreen = Toolkit.getDefaultToolkit().getScreenSize();
95      parent.setPreferredSize(fullScreen);
96      int width = (int) (fullScreen.getWidth());
97      int height = (int) (fullScreen.getHeight());
98      parent.setPreferredSize(new Dimension(width, height));
99  
100     parent.pack();
101     parent.setVisible(true);
102   }
