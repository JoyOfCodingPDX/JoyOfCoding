3   import com.google.common.eventbus.EventBus;
4   import com.google.common.eventbus.Subscribe;
5   import com.google.inject.Inject;
6   import com.google.inject.Singleton;
7   
8   import java.io.IOException;
9   import java.io.PrintWriter;
10  import java.io.StringWriter;
11  
12  @Singleton
13  public class UnhandledExceptionPresenter {
14    private final UnhandledExceptionView view;
15  
16    @Inject
17    public UnhandledExceptionPresenter(EventBus bus, UnhandledExceptionView view) {
18      this.view = view;
19      bus.register(this);
20    }
21  
22    @Subscribe
23    public void populateViewFrom(UnhandledExceptionEvent event) throws IOException {
24      Throwable exception = event.getUnhandledException();
25      this.view.setExceptionMessage(exception.getMessage());
26      this.view.setExceptionDetails(getStackTrace(exception));
27      this.view.displayView();
28    }
29  
30    private String getStackTrace(Throwable exception) throws IOException {
31      try (StringWriter sw = new StringWriter()) {
32        exception.printStackTrace(new PrintWriter(sw));
33        sw.flush();
34        return sw.toString();
35      }
36    }
37  }
