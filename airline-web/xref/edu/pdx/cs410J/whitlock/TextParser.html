5   import java.io.BufferedReader;
6   import java.io.IOException;
7   import java.io.Reader;
8   
9   public class TextParser {
10    private final Reader reader;
11  
12    public TextParser(Reader reader) {
13      this.reader = reader;
14    }
15  
16    public Airline parse() throws ParserException {
17      try (
18        BufferedReader br = new BufferedReader(this.reader)
19      ) {
20  
21        Airline airline = null;
22        for (String line = br.readLine(); line != null; line = br.readLine()) {
23          if (airline == null) {
24            airline = new Airline(line);
25  
26          } else {
27            airline.addFlight(new Flight(Integer.parseInt(line)));
28          }
29        }
30  
31        return airline;
32  
33      } catch (IOException e) {
34        throw new ParserException("While parsing dictionary", e);
35      }
36  
37    }
38  }
