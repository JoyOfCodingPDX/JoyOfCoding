12  public class PersonMain {
13    /**
14     * A simple test program.
15     */
16    public static void main(String[] args) {
17      // Make some people and fill in information
18      Person me = me();
19      Person dad = dad(me);
20      Person mom = mom(me);
21  
22      me.setMother(mom);
23      me.setFather(dad);
24  
25      // Print out descriptions of these people
26      System.out.println(me + "\n");
27      System.out.println(mom + "\n");
28      System.out.println(dad + "\n");
29  
30    }

32    /**

44    /**
45     * Returns a Person representing my mom.
46     */
47    static Person mom(Person me) {
48      Person mom = new Person(2, Person.FEMALE);
49      mom.setFirstName("Carolyn");
50      mom.setMiddleName("Joyce");
51      mom.setLastName("Granger");
52  
53      try {
54        DateFormat df = DateFormat.getDateInstance(DateFormat.MEDIUM);
55        mom.setDateOfBirth(df.parse("May 17, 1945"));
56  
57      } catch (ParseException ex) {
58        System.err.println("** Malformatted mom's birthday?");
59        System.exit(1);
60      }
61  
62      return mom;
63    }
