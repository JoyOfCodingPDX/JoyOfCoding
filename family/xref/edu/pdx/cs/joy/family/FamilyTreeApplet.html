3   import java.net.MalformedURLException;
4   import java.net.URL;
5   
6   import javax.swing.JApplet;
7   
8   /**
9    * This class is an applet that displays an immutable family tree.  
10   */
11  @SuppressWarnings("serial")
12  public class FamilyTreeApplet extends JApplet {
13  
14    /**
15     * Creates a <code>FamilyTreePanel</code> to view a family tree.
16     */
17    public void init() {
18      FamilyTreePanel viewer = new FamilyTreePanel();
19      viewer.addComponents();
20      this.getContentPane().add(viewer);
21  
22      String xmlFile = this.getParameter("xmlFile");
23      if (xmlFile != null) {
24        URL url = null;
25        try {
26  //         URL base = this.getDocumentBase();
27  //         url = new URL(base, xmlFile);
28          url = new URL(xmlFile);
29          viewer.setURLSource(url);
30  
31        } catch (MalformedURLException ex) {
32          viewer.setSourceText(ex.toString());
33        }
34      }
35    }
36  }
