001package edu.pdx.cs410J.gwt.client;
002
003import com.google.gwt.user.client.ui.HorizontalSplitPanel;
004import com.google.gwt.user.client.ui.Label;
005import com.google.gwt.user.client.ui.VerticalSplitPanel;
006
007/**
008 * Demonstrates GWT's
009 */
010public class SplitPanelExample extends Example {
011  public SplitPanelExample() {
012    super("Split Panels");
013
014    HorizontalSplitPanel horiz = new HorizontalSplitPanel();
015    horiz.setWidth("250px");
016    horiz.setHeight("250px");
017    horiz.setLeftWidget(new Label("Left"));
018    horiz.setRightWidget(new Label("Right"));
019//    horiz.setSplitPosition("10px");
020    add(horiz);
021
022    VerticalSplitPanel vert = new VerticalSplitPanel();
023    vert.setWidth("250px");
024    vert.setHeight("250px");
025    vert.setTopWidget(new Label("Top"));
026    vert.setBottomWidget(new Label("Bottom"));
027//    vert.setSplitPosition("15px");
028    add(vert);
029
030  }
031}
