001package edu.pdx.cs410J.apptbookweb;
002
003import java.io.PrintWriter;
004import java.util.HashMap;
005import java.util.Map;
006import java.util.regex.Matcher;
007import java.util.regex.Pattern;
008
009/**
010 * Class for formatting messages on the server side.  This is mainly to enable
011 * test methods that validate that the server returned expected strings.
012 */
013public class Messages
014{
015    public static String formatMappingCount(int count )
016    {
017        return String.format( "Server contains %d key/value pairs", count );
018    }
019
020    public static String formatKeyValuePair( String key, String value )
021    {
022        return String.format("  %s -> %s", key, value);
023    }
024
025    public static String missingRequiredParameter( String parameterName )
026    {
027        return String.format("The required parameter \"%s\" is missing", parameterName);
028    }
029
030    public static String mappedKeyValue( String key, String value )
031    {
032        return String.format( "Mapped %s to %s", key, value );
033    }
034
035    public static String allMappingsDeleted() {
036        return "All mappings have been deleted";
037    }
038
039    public static Map.Entry<String, String> parseKeyValuePair(String content) {
040        Pattern pattern = Pattern.compile("\\s*(.*) -> (.*)");
041        Matcher matcher = pattern.matcher(content);
042
043        if (!matcher.find()) {
044            return null;
045        }
046
047        return new Map.Entry<String, String>() {
048            @Override
049            public String getKey() {
050                return matcher.group(1);
051            }
052
053            @Override
054            public String getValue() {
055                String value = matcher.group(2);
056                if ("null".equals(value)) {
057                    value = null;
058                }
059                return value;
060            }
061
062            @Override
063            public String setValue(String value) {
064                throw new UnsupportedOperationException("This method is not implemented yet");
065            }
066        };
067    }
068
069    public static void formatKeyValueMap(PrintWriter pw, Map<String, String> map) {
070        pw.println(Messages.formatMappingCount(map.size()));
071
072        for (Map.Entry<String, String> entry : map.entrySet()) {
073            pw.println(Messages.formatKeyValuePair(entry.getKey(), entry.getValue()));
074        }
075    }
076
077    public static Map<String, String> parseKeyValueMap(String content) {
078        Map<String, String> map = new HashMap<>();
079
080        String[] lines = content.split("\n");
081        for (int i = 1; i < lines.length; i++) {
082            String line = lines[i];
083            Map.Entry<String, String> entry = parseKeyValuePair(line);
084            map.put(entry.getKey(), entry.getValue());
085        }
086
087        return map;
088    }
089}
