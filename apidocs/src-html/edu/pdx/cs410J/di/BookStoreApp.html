001package edu.pdx.cs410J.di;
002
003import javax.xml.bind.JAXBException;
004import java.io.File;
005import java.io.IOException;
006import java.util.Collections;
007import java.util.logging.Level;
008import java.util.logging.Logger;
009
010/**
011 * A main program that launches a book store using the "production" {@link edu.pdx.cs410J.di.BookInventory} and
012 * {@link edu.pdx.cs410J.di.CreditCardService}
013 */
014public class BookStoreApp
015{
016   public static void main(String... args) throws JAXBException, IOException
017   {
018       String tmpdir = System.getProperty( "java.io.tmpdir" );
019       File directory = new File( tmpdir );
020       BookInventory inventory = new BookDatabase( directory );
021       addBooks(inventory);
022       CreditCardService cardService = new FirstBankOfPSU( "localhost", 8080 );
023
024       Logger logger = Logger.getLogger( "edu.pdx.cs410J.Logger" );
025       logger.setLevel( Level.INFO );
026
027       CheckoutPanel panel = new CheckoutPanel( inventory, cardService, logger );
028       BookStoreGUI gui = new BookStoreGUI( panel );
029       gui.pack();
030       gui.setVisible( true );
031   }
032
033    protected static void addBooks( BookInventory inventory )
034    {
035        addBook( inventory, new Book( "The Pragmatic Programmer", "Andrew Hunt", 29.95 ), 4 );
036        addBook( inventory, new Book("Winning", "Jack Welch", 15.95), 3 );
037        addBook( inventory, new Book( "Agile Estimating and Planning", "Mike Cohn", 35.99), 2);
038    }
039
040    private static void addBook( BookInventory inventory, Book book, int copies )
041    {
042        inventory.add( Collections.nCopies( copies, book ).toArray(new Book[copies]) );
043    }
044}
