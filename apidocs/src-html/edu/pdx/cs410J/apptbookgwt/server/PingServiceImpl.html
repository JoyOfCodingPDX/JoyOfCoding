001package edu.pdx.cs410J.apptbookgwt.server;
002
003import com.google.gwt.user.server.rpc.RemoteServiceServlet;
004import edu.pdx.cs410J.apptbookgwt.client.Appointment;
005import edu.pdx.cs410J.apptbookgwt.client.AppointmentBook;
006import edu.pdx.cs410J.apptbookgwt.client.PingService;
007
008/**
009 * The server-side implementation of the division service
010 */
011public class PingServiceImpl extends RemoteServiceServlet implements PingService
012{
013  @Override
014  public AppointmentBook ping() {
015    AppointmentBook book = new AppointmentBook();
016    book.addAppointment(new Appointment());
017    return book;
018  }
019
020  @Override
021  protected void doUnexpectedFailure(Throwable unhandled) {
022    unhandled.printStackTrace(System.err);
023    super.doUnexpectedFailure(unhandled);
024  }
025
026}
