001package edu.pdx.cs410J.xml;
002
003import java.io.*;
004import java.util.*;
005import javax.xml.parsers.*;
006
007import org.w3c.dom.*;
008import org.xml.sax.*;
009
010/**
011 * This class represents a phone book that contains entries for
012 * businesses and residents.  It is constructed from an XML DOM tree.
013 * If we doing this for real, we'd want to be able to construct an
014 * empty <code>PhoneBook</code> and add entries.  But, come on, this
015 * is just an example.
016 */
017public class PhoneBook {
018  private static PrintStream err = System.err;
019
020  private Collection entries;
021
022  /**
023   * Creates a <code>PhoneBook</code> from an XML DOM tree.
024   */
025  public PhoneBook(Element root) {
026    // Verify that this is a phonebook XML doc
027    if (!root.getNodeName().equals("phonebook")) {
028      String s = "Not a phonebook: " + root.getNodeName();
029      throw new IllegalArgumentException(s);
030    }
031
032    this.entries = new ArrayList();
033    NodeList entries = root.getChildNodes();
034    for (int i = 0; i < entries.getLength(); i++) {
035      Node node = entries.item(i);
036      
037      if (!(node instanceof Element)) {
038        // Ignore other stuff
039        continue;
040      }
041
042      Element entry = (Element) node;
043
044      if (entry.getNodeName().equals("resident")) {
045        Resident resident = new Resident(entry);
046        this.entries.add(resident);
047
048      } else if (entry.getNodeName().equals("business")) {
049        Business business = new Business(entry);
050        this.entries.add(business);
051
052      } else {
053        String s = "Unknown entry: " + entry.getNodeName() + " (" +
054          entry.getNodeValue() + ")";
055        throw new IllegalArgumentException(s);
056      }
057    }
058  }
059
060  /**
061   * Test program that takes the name of a XML file from the command
062   * line and attempts to make a <code>PhoneBook</code> out of it.
063   */
064  public static void main(String[] args) {
065    // Parse the XML file to create a DOM tree
066    Document doc = null;
067    try {
068      DocumentBuilderFactory factory =
069        DocumentBuilderFactory.newInstance();
070      factory.setValidating(true);
071
072      DocumentBuilder builder = 
073        factory.newDocumentBuilder();
074      doc = builder.parse(new File(args[0]));
075
076    } catch (ParserConfigurationException ex) {
077      err.println("** " + ex);
078      System.exit(1);
079
080    } catch (SAXException ex) {
081      err.println("** SAXException: " + ex);
082      System.exit(1);
083
084    } catch (IOException ex) {
085      err.println("** IOException: " + ex);
086      System.exit(1);
087    }
088
089    Element root = (Element) doc.getChildNodes().item(1);
090    PhoneBook phonebook = new PhoneBook(root);
091    System.out.println(phonebook);
092  }
093
094  public String toString() {
095    StringBuffer sb = new StringBuffer();
096    sb.append("Phone Book\n\n");
097
098    Iterator iter = this.entries.iterator();
099    while (iter.hasNext()) {
100      sb.append(iter.next());
101      sb.append("\n");
102    }
103    
104    return sb.toString();
105  }
106}
