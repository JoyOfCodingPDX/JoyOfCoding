073   */
074  public Tester(final String id, File logDir) {
075    this.id = id;
076
077    if (!logDir.isDirectory()) {
078      throw new IllegalArgumentException(logDir + 
079                                         " is not a directory");
080    }
081
082    if (!logDir.exists()) {
083      logDir.mkdirs();
084    }
085
086    String logFileName = id + ".log";
087    File logFile = new File(logDir, logFileName);
088    
089
090    // Create a PrintStream around the log file.  Set System.out and
091    // System.err to refer to this guy.
092    FileOutputStream fos = null;
093    try {
094      fos = new FileOutputStream(logFile);
095
096    } catch (IOException ex) {
097      throw new IllegalArgumentException("Could not open " +
098                                         logFileName);
099    }

133  /**
134   * Parses a file path and creates an array of <code>URL</code>s
135   * representing each file on the path.
136   */
137  @SuppressWarnings("deprecation")
138public static URL[] parseURLPath(String path) {
139    if (path == null) {
140      return new URL[0];
141    }
142
143    StringTokenizer st = new StringTokenizer(path,
144                                             File.pathSeparator);
145    URL[] urls = new URL[st.countTokens()];
146    for (int i = 0; st.hasMoreTokens(); i++) {
147      File file = new File(st.nextToken());
148      try {
149        urls[i] = file.toURL();
150
151      } catch (MalformedURLException ex) {
152        System.err.println("** Malformatted URL: " + file);
153      }
154    }
155    
156    return urls;
157  }

178  /**
179   * Executes a static method of a given name in a given class with a
180   * given set of parameters.
181   *
182   * @param printBanner
183   *        Do we print a banner message?
184   */
185  public Object executeStatic(String className, String methodName, 
186                              Class[] paramTypes, Object[] params,
187                              boolean printBanner) {
188
189    ClassLoader loader = new URLClassLoader(this.urls);
190
191    // First load the class and the desired method
192    Class c = null;
193    try {
194      c = Class.forName(className, true, loader);
195
196    } catch (ClassNotFoundException ex) {
197      System.err.println("** Could not load " + className);
198      ex.printStackTrace(System.err);
199      return null;
200    }
201
202    Method m = null;
203    try {
204      m = c.getMethod(methodName, paramTypes);
205
206    } catch (NoSuchMethodException ex) {
207      System.err.println("** Could not find method " + methodName + 
208                         " in " + className);
209      ex.printStackTrace(System.err);
210      return null;
211    }
212
213    if (printBanner) {
214      // Print out some information
215      this.printBanner("Executing " + className + "." + methodName +
216                       "()", '-');
217    }
218
219    // Run the method
220    Object result = null;
221    try {
222      try {
223        result = m.invoke(null, params);
224
225      } catch (InvocationTargetException ex) {
226        throw ex.getTargetException();
227      }
228
229    } catch (TesterExitException ex) {
230      // That's okay, we're just exiting from one invocation of the
231      // main method
232
233    } catch (IllegalAccessException ex) {
234      System.err.println("** IllegalAccessException while invoking " +
235                         className + "." + methodName + "()");
236      ex.printStackTrace(System.err);
237      result = null;
238
239    } catch (Exception ex) {
240      // Log everything else
241      System.err.println("** Exception while invoking " +
242                         className + "." + methodName + "()");
243      ex.printStackTrace(System.err);
244      result = null;
245
246    } catch (Throwable ex) {
247      // EEEP!!!
248      System.err.println("*** SEVERE ERROR!!!");
249      ex.printStackTrace(System.err);
250      System.exit(1);
251    }
252
253    System.out.flush();
254    System.err.flush();
255
256    this.log.println("");
257    this.log.flush();
258    return result;
259  }

261  /**
262   * Executes the <code>main</code> method of the class with the given
263   * name in its own thread.
264   *
265   * @return The <code>Thread</code> in which the <code>main</code>
266   *         method is running.
267   */
268  public Thread executeMainInThread(final String className, 
269                                  final String[] args) {
270
271    this.printlnCentered("Executing " + className);
272
273    Runnable r = new Runnable() {
274        public void run() {
275          try {
276            Tester.this.executeMain(className, args);
277
278          } catch (Throwable t) {
279            return;
280          }
281        }
282      };
283    
284    Thread thread = new Thread(r);
285    thread.start();
286
287    return thread;
288  }

308  public void executeMain(String className, String[] args) {
309//      this.consoleOut.println("out: " + System.out);

312//      System.setOut(this.ps);

327    } catch (ClassNotFoundException ex) {
328      System.err.println("** Could not load " + className);

334    try {

347    // Process the arguments to replace any macros
348    args = expandMacros(args);

367      }

370    }

372   this.log.println("Command line: " + sb.toString().trim());
373    this.log.println("");
374    this.log.println("Program output:");
375
376    // Invoke the main method with the given arguments.
377    try {
378      Object[] actuals = {args};
379      
380      try {
381        main.invoke(null, actuals);

383      } catch (InvocationTargetException ex) {

407      ex.printStackTrace(System.err);

409    }
410//        this.consoleOut.println("Finally: " + args);
411//        System.out.println("Finally: " + args);
412    System.out.flush();

417  /**
418   * Go through an array of <code>String</code>s and expand macros
419   * such as "{STUDENT}".
420   *
421   * @return A clone of the <code>args</code> array with macros
422   *         expanded
423   */
424  String[] expandMacros(String[] args) {
425    Map<String, String> map = new HashMap<String, String>();
426    map.put("{STUDENT}", this.id);
427
428    args = (String[]) args.clone();
429
430    for (int i = 0; i < args.length; i++) {
431      Iterator keys = map.keySet().iterator();
432      while (keys.hasNext()) {
433        String key = (String) keys.next();
434        String value = (String) map.get(key);
435        int index = args[i].indexOf(key);
436        for(int j = 0; j < 20 && index != -1; j++) {
437          // Replace key with value
438          StringBuffer sb = new StringBuffer(args[i]);
439          sb.replace(index, index + key.length(), value);
440          args[i] = sb.toString();
441
442          index = args[i].indexOf(key);
443        }
444      }
445    }
446
447    return args;
448  }

471  /**
472   * Prints a line of text to the log file, centers it, around
473   * surrounds it with a given character.
474   */
475  public void printBanner(String text, char c) {
476    int length = text.length();
477
478    // Don't bother centering if the text is wider than the log.
479    if (length < WIDTH) {
480      int indent = (WIDTH - length) / 2;
481
482      for (int i = 0; i < indent - 1; i++) {
483        this.log.print(c);
484      }
485      this.log.print(' ');
486      this.log.flush();
487    }
488
489    this.log.print(text);
490
491    if (length < WIDTH) {
492      int indent = (WIDTH - length) / 2;
493
494      this.log.print(' ');
495      for (int i = 0; i < indent - 1; i++) {
496        this.log.print(c);
497      }
498      this.log.flush();
499    }
500
501    this.log.println("");
502  }

504  /**
505   * Copies one file to another.  If the destination file exists, it
506   * is overridden.
507   */
508  public void cp(File srcFile, File destFile) {
509    // Copy using streams in case we ever need to copy binary files.
510    FileInputStream src = null;
511    try {
512      src = new FileInputStream(srcFile);
513
514    } catch (FileNotFoundException ex) {
515      String s = "Could not find file " + srcFile;
516      throw new IllegalArgumentException(s);
517    }
518
519    try {
520      FileOutputStream dest = new FileOutputStream(destFile);
521      
522      byte[] buffer = new byte[1024];
523      int count = src.read(buffer);
524      while (count != -1) {
525        dest.write(buffer, 0, count);
526        count = src.read(buffer);
527      }
528      src.close();
529      dest.close();
530
531    } catch (IOException ex) {
532      String s = "IOException " + ex;
533      throw new IllegalArgumentException(s);
534    }
535  }

537  /**
538   * "Cats" a file to the log.  That is, copies its contents to the
539   * log.
540   *
541   * @param file
542   *        <code>File</code> to be catted
543   */
544  public void cat(File file) {
545    printlnCentered("File: " + file.getName());
546    this.log.println("");
547
548    if (!file.exists()) {
549      printlnCentered("Does not exist!");
550      return;
551    }
552
553    try {
554      FileReader fr = new FileReader(file);
555      BufferedReader br = new BufferedReader(fr);
556
557      while (br.ready()) {
558        String line = br.readLine();
559        this.log.println(line);
560      }
561
562    } catch (FileNotFoundException ex) {
563      String s = "Could not find file: " + file;
564      throw new IllegalArgumentException(s);
565
566    } catch (IOException ex) {
567      throw new IllegalArgumentException(ex.toString());
568    }
569
570    this.log.println("\n");
571  }
