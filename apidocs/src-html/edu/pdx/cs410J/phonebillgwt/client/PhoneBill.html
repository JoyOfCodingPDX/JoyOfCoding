001package edu.pdx.cs410J.phonebillgwt.client;
002
003import edu.pdx.cs410J.AbstractPhoneBill;
004
005import java.lang.Override;
006import java.util.ArrayList;
007import java.util.Collection;
008
009public class PhoneBill extends AbstractPhoneBill<PhoneCall> {
010  private Collection<PhoneCall> calls = new ArrayList<>();
011
012  /**
013   * In order for GWT to serialize this class (so that it can be sent between
014   * the client and the server), it must have a zero-argument constructor.
015   */
016  public PhoneBill() {
017
018  }
019
020  @Override
021  public String getCustomer() {
022    return "CS410J";
023  }
024
025  @Override
026  public void addPhoneCall(PhoneCall call) {
027    this.calls.add(call);
028  }
029
030  @Override
031  public Collection<PhoneCall> getPhoneCalls() {
032    return this.calls;
033  }
034}
