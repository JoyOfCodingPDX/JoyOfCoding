001package edu.pdx.cs410J.web;
002
003import java.net.HttpURLConnection;
004import java.net.URL;
005import java.net.URLEncoder;
006import java.io.*;
007import java.nio.charset.Charset;
008
009/**
010 * A helper class that provides methods for requesting resources via HTTP
011 */
012public class HttpRequestHelper {
013  /**
014   * Performs an HTTP GET on the given URL
015   *
016   * @param urlString The URL to get
017   * @param parameters The key/value query parameters
018   * @return A <code>Response</code> summarizing the result of the GET
019   */
020  protected Response get(String urlString, String... parameters) throws IOException {
021    checkParameters(parameters);
022
023    StringBuilder query = encodeParameters(parameters);
024    if (query.length() > 0) {
025      query.insert(0, '?');
026    }
027
028    URL url = new URL(urlString + query);
029    HttpURLConnection conn = (HttpURLConnection) url.openConnection();
030    conn.setRequestMethod("GET");
031    conn.setDoOutput(true);
032    conn.setDoInput(true);
033
034    return new Response(conn);
035
036  }
037
038  /**
039   * Performs an HTTP POST to the given URL
040   * @param urlString The URL to post to
041   * @param parameters The key/value parameters
042   * @return A <code>Response</code> summarizing the result of the POST
043   */
044  protected Response post(String urlString, String... parameters) throws IOException {
045    return sendEncodedRequest(urlString, "POST", parameters);
046
047  }
048
049  /**
050   * Performs an HTTP DELETE to the given URL
051   * @param urlString The URL to post to
052   * @param parameters The key/value parameters
053   * @return A <code>Response</code> summarizing the result of the POST
054   */
055  protected Response delete(String urlString, String... parameters) throws IOException {
056    return sendEncodedRequest(urlString, "DELETE", parameters);
057
058  }
059
060  private Response sendEncodedRequest(String urlString, String requestMethod, String... parameters) throws IOException {
061    checkParameters(parameters);
062
063    StringBuilder data = encodeParameters(parameters);
064
065    URL url = new URL(urlString);
066    HttpURLConnection conn = (HttpURLConnection) url.openConnection();
067    conn.setRequestMethod(requestMethod);
068    conn.setDoOutput(true);
069
070    OutputStreamWriter wr = new OutputStreamWriter(conn.getOutputStream(), Charset.forName("UTF-8"));
071    wr.write(data.toString());
072    wr.flush();
073
074    Response response = new Response(conn);
075    wr.close();
076
077    return response;
078  }
079
080  /**
081   * Encodes parameters to be sent to the server via an HTTP GET, POST, or DELETE
082   * @param parameters The parameter key/value pairs
083   * @return The encoded parameters
084   * @throws java.io.UnsupportedEncodingException If we can't encode UTF-8
085   */
086  private StringBuilder encodeParameters(String... parameters) throws UnsupportedEncodingException {
087    StringBuilder query = new StringBuilder();
088    for (int i = 0; i < parameters.length; i += 2) {
089      String key = parameters[i];
090      String value = parameters[i + 1];
091      query.append(URLEncoder.encode(key, "UTF-8"));
092      query.append("=");
093      query.append(URLEncoder.encode(value, "UTF-8"));
094      if (i < parameters.length - 2) {
095        query.append("&");
096      }
097    }
098    return query;
099  }
100
101  private void checkParameters(String... parameters) {
102    if (parameters.length % 2 != 0) {
103      String s = "You must specify an even number of parameters (key/value pairs)";
104      throw new IllegalArgumentException(s);
105    }
106  }
107
108  /**
109   * Performs an HTTP PUT on the given URL
110   *
111   * @param urlString The URL to put to
112   * @param parameters key/value parameters to the put
113   * @return A <code>Reponse</code> summarizing the result of the PUT
114   */
115  protected Response put(String urlString, String... parameters) throws IOException {
116    checkParameters(parameters);
117
118    StringBuilder data = new StringBuilder();
119    for (int i = 0; i < parameters.length; i += 2) {
120      String key = parameters[i];
121      String value = parameters[i + 1];
122      data.append(key);
123      data.append("=");
124      data.append(value);
125      data.append("\n");
126    }
127
128    URL url = new URL(urlString);
129    HttpURLConnection conn = (HttpURLConnection) url.openConnection();
130    conn.setRequestMethod("PUT");
131    conn.setDoOutput(true);
132    conn.setDoInput(true);
133    conn.setRequestProperty("Content-Type", "text/plain");
134    conn.setRequestProperty("Context-Length", String.valueOf(data.length()));
135
136    OutputStreamWriter wr = new OutputStreamWriter(conn.getOutputStream(), Charset.forName("UTF-8"));
137    wr.write(data.toString());
138    wr.flush();
139
140    Response response = new Response(conn);
141    wr.close();
142
143    return response;
144  }
145
146  /**
147   * Encapsulates a response to an HTTP request
148   */
149  public static class Response {
150
151    private final int code;
152
153    private final String content;
154
155    private int contentLines = 0;
156
157    private Response(HttpURLConnection conn) throws IOException {
158      this.code = conn.getResponseCode();
159      if (this.code != java.net.HttpURLConnection.HTTP_OK) {
160        this.content = conn.getResponseMessage();
161        return;
162      }
163
164      StringBuilder content = new StringBuilder();
165      BufferedReader rd = new BufferedReader(new InputStreamReader(conn.getInputStream()));
166      String line;
167      while ((line = rd.readLine()) != null) {
168        content.append(line);
169        content.append("\n");
170        contentLines++;
171      }
172      rd.close();
173
174      this.content = content.toString().trim();
175    }
176
177    /**
178     * Returns the HTTP status code of the response
179     * @see java.net.HttpURLConnection#getResponseCode()
180     */
181    public int getCode() {
182      return code;
183    }
184
185    /**
186     * Returns the (presumably textual) response from the URL
187     * @return
188     */
189    public String getContent() {
190      return content;
191    }
192
193    /**
194     * Returns the number of lines in the response's content
195     */
196    public int getContentLines() {
197      return contentLines;
198    }
199  }
200
201  /**
202   * A main method that requests a resource from a URL using a given HTTP method
203   */
204  public static void main(String[] args) throws IOException {
205    String method = args[0];
206    String url = args[1];
207    String[] parameters = new String[args.length - 2];
208    System.arraycopy(args, 2, parameters, 0, parameters.length);
209
210    HttpRequestHelper helper = new HttpRequestHelper();
211
212    Response response;
213    if (method.equalsIgnoreCase("PUT")) {
214      response = helper.put(url, parameters);
215
216    } else if (method.equalsIgnoreCase("GET")) {
217      response = helper.get(url, parameters);
218
219    } else if (method.equalsIgnoreCase("POST")) {
220      response = helper.post(url, parameters);
221
222    } else {
223      System.err.println("** Unknown method: " + method);
224      return;
225    }
226
227    System.out.println("Returned code " + response.getCode() + " and " +
228      response.getContentLines() + " lines of content\n");
229    System.out.println(response.getContent());
230
231  }
232}
