003import java.io.*;
004import java.net.HttpURLConnection;
005import java.net.URL;
006import java.net.URLEncoder;
007import java.nio.charset.Charset;
008import java.nio.charset.StandardCharsets;
009import java.util.HashMap;
010import java.util.Iterator;
011import java.util.Map;
012
013/**
014 * A helper class that provides methods for requesting resources via HTTP
015 */
016public class HttpRequestHelper {
017
018  /**
019   * Performs an HTTP GET on the given URL
020   *
021   * @param urlString  The URL to get
022   * @param parameters The key/value query parameters
023   * @return A <code>Response</code> summarizing the result of the GET
024   */
025  protected Response get(String urlString, Map<String, String> parameters) throws IOException {
026    StringBuilder query = encodeParameters(parameters);
027    if (query.length() > 0) {
028      query.insert(0, '?');
029    }
030
031    URL url = new URL(urlString + query);
032    HttpURLConnection conn = (HttpURLConnection) url.openConnection();
033    conn.setRequestMethod("GET");
034    conn.setDoOutput(true);
035    conn.setDoInput(true);
036
037    return new Response(conn);
038
039  }
040
041  /**
042   * Performs an HTTP POST to the given URL
043   * @param urlString The URL to post to
044   * @param parameters The key/value parameters
045   * @return A <code>Response</code> summarizing the result of the POST
046   */
047  protected Response post(String urlString, Map<String, String> parameters) throws IOException {
048    return sendEncodedRequest(urlString, "POST", parameters);
049  }
050
051  /**
052   * Performs an HTTP DELETE to the given URL
053   * @param urlString The URL to post to
054   * @param parameters The key/value parameters
055   * @return A <code>Response</code> summarizing the result of the POST
056   */
057  protected Response delete(String urlString, Map<String, String> parameters) throws IOException {
058    return sendEncodedRequest(urlString, "DELETE", parameters);
059  }
060
061  private Response sendEncodedRequest(String urlString, String requestMethod, Map<String, String> parameters) throws IOException {
062    StringBuilder data = encodeParameters(parameters);
063
064    URL url = new URL(urlString);
065    HttpURLConnection conn = (HttpURLConnection) url.openConnection();
066    conn.setRequestMethod(requestMethod);
067    conn.setDoOutput(true);
068
069    OutputStreamWriter wr = new OutputStreamWriter(conn.getOutputStream(), Charset.forName("UTF-8"));
070    wr.write(data.toString());
071    wr.flush();
072
073    Response response = new Response(conn);
074    wr.close();
075
076    return response;
077  }
078
079  /**
080   * Encodes parameters to be sent to the server via an HTTP GET, POST, or DELETE
081   * @param parameters The parameter key/value pairs
082   * @return The encoded parameters
083   * @throws java.io.UnsupportedEncodingException If we can't encode UTF-8
084   */
085  private StringBuilder encodeParameters(Map<String, String> parameters) throws UnsupportedEncodingException {
086    StringBuilder query = new StringBuilder();
087    for (Iterator<Map.Entry<String, String>> iter = parameters.entrySet().iterator(); iter.hasNext(); ) {
088      Map.Entry<String, String> pair = iter.next();
089      String key = pair.getKey();
090      String value = pair.getValue();
091      query.append(URLEncoder.encode(key, StandardCharsets.UTF_8));
092      query.append("=");
093      query.append(URLEncoder.encode(value, StandardCharsets.UTF_8));
094      if (iter.hasNext()) {
095        query.append("&");
096      }
097    }
098    return query;
099  }
100
101  /**
102   * Performs an HTTP PUT on the given URL
103   *
104   * @param urlString The URL to put to
105   * @param parameters key/value parameters to the put
106   * @return A <code>Response</code> summarizing the result of the PUT
107   */
108  protected Response put(String urlString, Map<String, String> parameters) throws IOException {
109    StringBuilder data = new StringBuilder();
110        parameters.forEach((key, value) -> {
111          data.append(key);
112          data.append("=");
113          data.append(value);
114          data.append("\n");
115        });
116
117    URL url = new URL(urlString);
118    HttpURLConnection conn = (HttpURLConnection) url.openConnection();
119    conn.setRequestMethod("PUT");
120    conn.setDoOutput(true);
121    conn.setDoInput(true);
122    conn.setRequestProperty("Content-Type", "text/plain");
123    conn.setRequestProperty("Context-Length", String.valueOf(data.length()));
124
125    OutputStreamWriter wr = new OutputStreamWriter(conn.getOutputStream(), Charset.forName("UTF-8"));
126    wr.write(data.toString());
127    wr.flush();
128
129    Response response = new Response(conn);
130    wr.close();
131
132    return response;
133  }
134
135  /**
136   * Encapsulates a response to an HTTP request
137   */
138  public static class Response {
139
140    private final int code;
141
142    private final String content;
143
144    private int contentLines = 0;
145
146    private Response(HttpURLConnection conn) throws IOException {
147      this.code = conn.getResponseCode();
148      if (this.code != java.net.HttpURLConnection.HTTP_OK) {
149        this.content = conn.getResponseMessage();
150        return;
151      }
152
153      StringBuilder content = new StringBuilder();
154      BufferedReader rd = new BufferedReader(new InputStreamReader(conn.getInputStream()));
155      String line;
156      while ((line = rd.readLine()) != null) {
157        content.append(line);
158        content.append("\n");
159        contentLines++;
160      }
161      rd.close();
162
163      this.content = content.toString().trim();
164    }
165
166    /**
167     * Returns the HTTP status code of the response
168     * @see java.net.HttpURLConnection#getResponseCode()
169     */
170    public int getCode() {
171      return code;
172    }
173
174    /**
175     * Returns the (presumably textual) response from the URL
176     */
177    public String getContent() {
178      return content;
179    }
180
181    /**
182     * Returns the number of lines in the response's content
183     */
184    public int getContentLines() {
185      return contentLines;
186    }
187  }
188
189
190  private static Map<String, String> arrayToMap(String[] parameters) {
191    Map<String, String> params = new HashMap<>();
192    for (int i = 0; i < parameters.length; i++) {
193      String key = parameters[i];
194      i++;
195      String value = parameters[i];
196      params.put(key, value);
197    }
198    return params;
199  }
200
201  /**
202   * A main method that requests a resource from a URL using a given HTTP method
203   */
204  public static void main(String[] args) throws IOException {
205    String method = args[0];
206    String url = args[1];
207    String[] parameters = new String[args.length - 2];
208    System.arraycopy(args, 2, parameters, 0, parameters.length);
209    Map<String, String> map = HttpRequestHelper.arrayToMap(parameters);
210
211    HttpRequestHelper helper = new HttpRequestHelper();
212
213    Response response;
214    if (method.equalsIgnoreCase("PUT")) {
215      response = helper.put(url, map);
216
217    } else if (method.equalsIgnoreCase("GET")) {
218      response = helper.get(url, map);
219
220    } else if (method.equalsIgnoreCase("POST")) {
221      response = helper.post(url, map);
222
223    } else {
224      System.err.println("** Unknown method: " + method);
225      return;
226    }
227
228    System.out.println("Returned code " + response.getCode() + " and " +
229      response.getContentLines() + " lines of content\n");
230    System.out.println(response.getContent());
231
232  }
