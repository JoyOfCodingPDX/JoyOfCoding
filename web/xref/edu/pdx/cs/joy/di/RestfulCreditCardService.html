1   package edu.pdx.cs.joy.di;
2   
3   import com.google.common.collect.ImmutableMap;
4   import com.google.inject.Inject;
5   import jakarta.ws.rs.*;
6   
7   import java.util.Map;
8   
9   /**
10   * A Credit Card service that exposes its behavior via REST
11   */
12  @Path("/rest/creditCard")
13  public class RestfulCreditCardService
14  {
15      private final CreditCardDatabase cards;
16  
17      @Inject
18      public RestfulCreditCardService(CreditCardDatabase cards) {
19          this.cards = cards;
20      }
21  
22      @GET
23      @Produces("text/plain")
24      public String allCardBalances() {
25          StringBuilder sb = new StringBuilder();
26  
27          ImmutableMap<CreditCard, Double> balances = cards.getBalances();
28          for ( Map.Entry<CreditCard, Double> entry : balances.entrySet() ) {
29              sb.append(entry.getKey().getNumber());
30              sb.append("\t");
31              sb.append(entry.getValue());
32              sb.append("\n");
33          }
34  
35          return sb.toString();
36      }
37  
38  }
