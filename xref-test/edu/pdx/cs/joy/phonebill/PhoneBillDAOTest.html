3   import org.junit.jupiter.api.AfterEach;
4   import org.junit.jupiter.api.BeforeEach;
5   import org.junit.jupiter.api.Test;
6   
7   import java.sql.*;
8   
9   import static org.hamcrest.MatcherAssert.assertThat;
10  import static org.hamcrest.Matchers.*;
11  
12  /**
13   * A simple example unit test that demonstrates persisting a PhoneBill
14   * to an H2 in-memory database using JDBC.
15   *
16   * This is a starting point for students to understand how to use
17   * Data Access Objects (DAOs) to persist domain objects to a database.
18   */
19  public class PhoneBillDAOTest {
20  
21    private Connection connection;
22    private PhoneBillDAO dao;
23  
24    @BeforeEach
25    public void setUp() throws SQLException {
26      // Create an in-memory H2 database
27      connection = DriverManager.getConnection("jdbc:h2:mem:phonebill_test");
28  
29      // Create the phone_bills table
30      PhoneBillDAO.createTable(connection);
31  
32      // Create the DAO
33      dao = new PhoneBillDAO(connection);
34    }
35  
36    @AfterEach
37    public void tearDown() throws SQLException {
38      if (connection != null && !connection.isClosed()) {
39        connection.close();
40      }
41    }
42  
43    @Test
44    public void canPersistAndFetchPhoneBillByCustomerName() throws SQLException {
45      String customerName = "Jane Doe";
46      PhoneBill bill = new PhoneBill(customerName);
47  
48      // Persist the phone bill using the DAO
49      dao.save(bill);
50  
51      // Fetch the phone bill by customer name
52      PhoneBill fetchedBill = dao.findByCustomer(customerName);
53  
54      // Validate that the fetched bill matches the original
55      assertThat(fetchedBill, is(notNullValue()));
56      assertThat(fetchedBill.getCustomer(), is(equalTo(customerName)));
57    }
58  
59    @Test
60    public void returnsNullWhenPhoneBillNotFound() throws SQLException {
61      PhoneBill fetchedBill = dao.findByCustomer("Non-existent Customer");
62      assertThat(fetchedBill, is(nullValue()));
63    }
64  }
