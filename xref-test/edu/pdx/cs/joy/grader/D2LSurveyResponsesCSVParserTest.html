12  public class D2LSurveyResponsesCSVParserTest {
13  
14    @Test
15    public void canParseFirstLineOfCSV() throws IOException {
16      Reader reader = createReaderWithLines("Section #,Q #,Q Type,Q Title,Q Text,Bonus?,Difficulty,Answer,Answer Match,# Responses");
17      D2LSurveyResponsesCSVParser parser = new D2LSurveyResponsesCSVParser(reader);
18  
19      assertThat(parser.questionColumnIndex, equalTo(4));
20      assertThat(parser.responseColumnIndex, equalTo(7));
21  
22    }
23  
24    @Test
25    public void canParseOneResponseLine() throws IOException {
26      String question = "What is your response?";
27      String response = "This is my response";
28      Reader reader = createReaderWithLines("Q Text,Answer", question + "," + response);
29  
30      D2LSurveyResponsesCSVParser parser = new D2LSurveyResponsesCSVParser(reader);
31      SurveyResponsesFromD2L responses = parser.getSurveyResponses();
32      assertThat(responses.getResponsesTo(question), hasItem(response));
33    }
34  
35    @Test
36    public void canParseTwoResponseLines() throws IOException {
37      String question = "What is your response?";
38      String response1 = "This is my response";
39      String response2 = "This is my other response";
40      Reader reader = createReaderWithLines("Q Text,Answer", question + "," + response1, question + "," + response2);
41  
42      D2LSurveyResponsesCSVParser parser = new D2LSurveyResponsesCSVParser(reader);
43      SurveyResponsesFromD2L responses = parser.getSurveyResponses();
44      assertThat(responses.getResponsesTo(question), hasItems(response1, response2));
45    }
46  
47    @Test
48    public void canParseMultipleQuestion() throws IOException {
49      String question1 = "This is my first question?";
50      String response1 = "This is my response";
51      String question2 = "This is my second question?";
52      String response2 = "This is my other response";
53      Reader reader = createReaderWithLines("Q Text,Answer", question1 + "," + response1, question2 + "," + response2);
54  
55      D2LSurveyResponsesCSVParser parser = new D2LSurveyResponsesCSVParser(reader);
56      SurveyResponsesFromD2L responses = parser.getSurveyResponses();
57      assertThat(responses.getResponsesTo(question1), hasItems(response1));
58      assertThat(responses.getResponsesTo(question2), hasItems(response2));
59    }
60  
61    private Reader createReaderWithLines(String... lines) {
62      StringWriter sw = new StringWriter();
63      PrintWriter pw = new PrintWriter(sw);
64      for (String line : lines) {
65        pw.println(line);
66      }
67  
68      return new StringReader(sw.toString());
69    }
