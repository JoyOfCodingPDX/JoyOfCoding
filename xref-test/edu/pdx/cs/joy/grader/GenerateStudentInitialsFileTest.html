16  public class GenerateStudentInitialsFileTest {
17  
18    @Test
19    void generatedInitialsAreSortedByLastNameInitial() {
20      Stream<Student> students = Stream.of(
21        new Student("larry").setFirstName("Larry").setLastName("Smith"),
22      new Student("zoe").setFirstName("Zoe").setLastName("Adams"),
23        new Student("alice").setFirstName("Alice").setLastName("Smith")
24      );
25  
26      List<String> initials = GenerateStudentInitialsFile.generateInitials(students);
27  
28      assertThat(initials, hasSize(3));
29  
30      Iterator<String> iterator = initials.iterator();
31      assertThat(iterator.next(), equalTo("ZA"));
32      assertThat(iterator.next(), equalTo("AS"));
33      assertThat(iterator.next(), equalTo("LS"));
34    }
35  
36    @Test
37    void onlyFirstLetterOfNameIsCapitalized() {
38      StudentInitials initials = new StudentInitials(new Student("test").setFirstName("Aa").setLastName("Bb"));
39      initials.incrementUniqueCharsOfFirstName();
40      assertThat(initials.getInitials(), equalTo("AaB"));
41      initials.incrementUniqueCharsOfLastName();
42      assertThat(initials.getInitials(), equalTo("AaBb"));
43    }
44  
45    @Test
46    void compareStudentInitials() {
47      StudentInitials first = new StudentInitials(new Student("first").setFirstName("A").setLastName("A"));
48      StudentInitials second = new StudentInitials(new Student("second").setFirstName("A").setLastName("B"));
49      assertThat(first.compareTo(second), equalTo(-1));
50      assertThat(second.compareTo(first), equalTo(1));
51      assertThat(first.getInitials(), equalTo("AA"));
52      assertThat(second.getInitials(), equalTo("AB"));
53  
54      first = new StudentInitials(new Student("first").setFirstName("A").setLastName("Aa"));
55      second = new StudentInitials(new Student("second").setFirstName("A").setLastName("Ab"));
56      assertThat(first.compareTo(second), equalTo(-1));
57      assertThat(second.compareTo(first), equalTo(1));
58      assertThat(first.getInitials(), equalTo("AAa"));
59      assertThat(second.getInitials(), equalTo("AAb"));
60    }
61  
62    @Test
63    void useMoreCharactersWhenInitialsMatch() {
64      Stream<Student> students = Stream.of(
65        new Student("zoe").setFirstName("Zoe").setLastName("Adams"),
66        new Student("zelda").setFirstName("Zelda").setLastName("Adams"),
67        new Student("larry").setFirstName("Larry").setLastName("Johnson"),
68        new Student("larry").setFirstName("Larry").setLastName("Jones")
69      );
70  
71      List<String> initials = GenerateStudentInitialsFile.generateInitials(students);
72  
73      assertThat(initials, hasSize(4));
74  
75      Iterator<String> iterator = initials.iterator();
76      assertThat(iterator.next(), equalTo("ZeAd"));
77      assertThat(iterator.next(), equalTo("ZoAd"));
78      assertThat(iterator.next(), equalTo("LaJoh"));
79      assertThat(iterator.next(), equalTo("LaJon"));
80  
81    }
