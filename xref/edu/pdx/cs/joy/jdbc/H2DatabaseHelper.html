1   package edu.pdx.cs.joy.jdbc;

3   import java.io.File;
4   import java.sql.Connection;
5   import java.sql.DriverManager;
6   import java.sql.SQLException;
7   
8   /**
9    * Helper class for creating connections to H2 databases.
10   * Provides factory methods for both in-memory and file-based H2 databases.
11   */
12  public class H2DatabaseHelper {
13  
14    /**
15     * Creates a connection to an in-memory H2 database.
16     * The database will persist as long as at least one connection remains open
17     * due to the DB_CLOSE_DELAY=-1 parameter.
18     *
19     * @param databaseName the name of the in-memory database
20     * @return a connection to the in-memory H2 database
21     * @throws SQLException if a database error occurs
22     */
23    public static Connection createInMemoryConnection(String databaseName) throws SQLException {
24      return DriverManager.getConnection("jdbc:h2:mem:" + databaseName + ";DB_CLOSE_DELAY=-1");
25    }
26  
27    /**
28     * Creates a connection to a file-based H2 database.
29     * The database will be persisted to a file at the specified path.
30     *
31     * @param databaseFilesDirectory the database file (without the .mv.db extension)
32     * @return a connection to the file-based H2 database
33     * @throws SQLException if a database error occurs
34     */
35    public static Connection createFileBasedConnection(File databaseFilesDirectory) throws SQLException {
36      return DriverManager.getConnection("jdbc:h2:" + databaseFilesDirectory.getAbsolutePath());
37    }
38  }
