3   import java.io.File;
4   import java.sql.Connection;
5   import java.sql.SQLException;
6   import java.util.List;

8   /**
9    * A command-line program that demonstrates CRUD operations on Department objects
10   * using the DepartmentDAO class with an H2 database.
11   */
12  public class ManageDepartments {
13  
14    /**
15     * Main method that performs CRUD operations on departments in an H2 database file.
16     *
17     * @param args command line arguments where args[0] is the path to the database file
18     *             and args[1] is the command (create, retrieve, update, delete, or list)
19     * @throws SQLException if a database error occurs
20     */
21    public static void main(String[] args) throws SQLException {
22      if (args.length < 2) {
23        printUsage();
24        return;
25      }
26  
27      String dbFilePath = args[0];
28      String command = args[1].toLowerCase();
29  
30      File dbFile = new File(dbFilePath);
31  
32      try (Connection connection = H2DatabaseHelper.createFileBasedConnection(dbFile)) {
33        // Create the departments table if it doesn't exist
34        DepartmentDAOImpl.createTable(connection);
35  
36        // Create a new DepartmentDAO
37        DepartmentDAO departmentDAO = new DepartmentDAOImpl(connection);
38  
39        switch (command) {
40          case "create":
41            handleCreate(args, departmentDAO);
42            break;
43          case "retrieve":
44            handleRetrieve(args, departmentDAO);
45            break;
46          case "update":
47            handleUpdate(args, departmentDAO);
48            break;
49          case "delete":
50            handleDelete(args, departmentDAO);
51            break;
52          case "list":
53            handleList(departmentDAO);
54            break;
55          default:
56            System.err.println("Unknown command: " + command);
57            printUsage();
58        }
59      }
60    }
61  
62    private static void printUsage() {
63      System.err.println("Usage: java ManageDepartments <database-file-path> <command> [args...]");
64      System.err.println();
65      System.err.println("Commands:");
66      System.err.println("  create <name>           - Create a new department with the given name");
67      System.err.println("  retrieve <id>           - Retrieve a department by ID");
68      System.err.println("  update <id> <name>      - Update the name of a department");
69      System.err.println("  delete <id>             - Delete a department by ID");
70      System.err.println("  list                    - List all departments");
71    }
72  
73    private static void handleCreate(String[] args, DepartmentDAO departmentDAO) throws SQLException {
74      if (args.length < 3) {
75        System.err.println("Missing department name for create command");
76        System.err.println("Usage: java ManageDepartments <database-file-path> create <name>");
77        return;
78      }
79  
80      String departmentName = args[2];
81      Department department = new Department(departmentName);
82      departmentDAO.save(department);
83  
84      System.out.println("Successfully created department:");
85      System.out.println(department);
86      System.out.println("Auto-generated ID: " + department.getId());
87    }
88  
89    private static void handleRetrieve(String[] args, DepartmentDAO departmentDAO) throws SQLException {
90      if (args.length < 3) {
91        System.err.println("Missing department ID for retrieve command");
92        System.err.println("Usage: java ManageDepartments <database-file-path> retrieve <id>");
93        return;
94      }
95  
96      int id = Integer.parseInt(args[2]);
97      Department department = departmentDAO.findById(id);
98  
99      if (department == null) {
100       System.out.println("No department found with ID: " + id);
101     } else {
102       System.out.println("Found department:");
103       System.out.println(department);
104     }
105   }
106 
107   private static void handleUpdate(String[] args, DepartmentDAO departmentDAO) throws SQLException {
108     if (args.length < 4) {
109       System.err.println("Missing arguments for update command");
110       System.err.println("Usage: java ManageDepartments <database-file-path> update <id> <name>");
111       return;
112     }
113 
114     int id = Integer.parseInt(args[2]);
115     String newName = args[3];
116 
117     Department department = departmentDAO.findById(id);
118     if (department == null) {
119       System.out.println("No department found with ID: " + id);
120       return;
121     }
122 
123     department.setName(newName);
124     departmentDAO.update(department);
125 
126     System.out.println("Successfully updated department:");
127     System.out.println(department);
128   }
129 
130   private static void handleDelete(String[] args, DepartmentDAO departmentDAO) throws SQLException {
131     if (args.length < 3) {
132       System.err.println("Missing department ID for delete command");
133       System.err.println("Usage: java ManageDepartments <database-file-path> delete <id>");
134       return;
135     }
136 
137     int id = Integer.parseInt(args[2]);
138     Department department = departmentDAO.findById(id);
139 
140     if (department == null) {
141       System.out.println("No department found with ID: " + id);
142       return;
143     }
144 
145     departmentDAO.delete(id);
146     System.out.println("Successfully deleted department:");
147     System.out.println(department);
148   }
149 
150   private static void handleList(DepartmentDAO departmentDAO) throws SQLException {
151     List<Department> allDepartments = departmentDAO.findAll();
152     System.out.println("Found " + allDepartments.size() + " department(s)");
153     for (Department dept : allDepartments) {
154       System.out.println("  " + dept);
155     }
156   }
