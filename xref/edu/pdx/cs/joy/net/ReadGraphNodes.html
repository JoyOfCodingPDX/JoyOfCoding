15    public static void main(String[] args) {

17  
18      try {
19        FileInputStream fis = new FileInputStream(fileName);
20        ObjectInputStream in = new ObjectInputStream(fis);
21        GraphNode root = (GraphNode) in.readObject();
22        System.out.println("Graph has " + root.traverse() + " nodes");
23  
24      } catch (ClassNotFoundException ex) {

26        System.exit(1);
