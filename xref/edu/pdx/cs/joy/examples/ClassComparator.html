8   /**
9    * This class is a <code>Comparator</code> that compares
10   * <code>Object</code>s based on the name of their classes.
11   */
12  public class ClassComparator implements Comparator<Object> {
13    public int compare(Object o1, Object o2) {
14      String name1 = o1.getClass().getName();
15      String name2 = o2.getClass().getName();
16      
17      // Take advantage of String's compartTo method
18      return name1.compareTo(name2);
19    }
20  
21    /**
22     * Create a bunch of object and store them in a set using a
23     * <code>ClassComparator</code>. 
24     */
25    public static void main(String[] args) {
26      Set<Object> set = new TreeSet<Object>(new ClassComparator());
27      set.add("Hello");
28      set.add(new Cow("Betty"));
29      set.add(new Vector());
30      set.add(new Ant("Richard"));
31  
32      // Print out the set
33      Iterator iter = set.iterator();
34      while (iter.hasNext()) {
35        Object o = iter.next();
36        String s = o + " (" + o.getClass().getName() + ")";
37        System.out.println(s);
38      }
39    }
40  }
