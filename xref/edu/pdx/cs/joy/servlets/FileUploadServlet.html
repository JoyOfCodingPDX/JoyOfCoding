3   import jakarta.servlet.ServletException;
4   import jakarta.servlet.http.HttpServlet;
5   import jakarta.servlet.http.HttpServletRequest;
6   import jakarta.servlet.http.HttpServletResponse;
7   import org.apache.commons.fileupload2.core.DiskFileItem;
8   import org.apache.commons.fileupload2.core.DiskFileItemFactory;
9   import org.apache.commons.fileupload2.core.FileUploadException;
10  import org.apache.commons.fileupload2.jakarta.servlet6.JakartaServletFileUpload;
11  
12  import java.io.BufferedReader;
13  import java.io.IOException;
14  import java.io.InputStreamReader;
15  import java.io.PrintWriter;
16  import java.util.List;
17  
18  /**
19   * A servlet that demonstrates how to upload a file to a web server.  It uses the Apache Commons
20   * <a href="http://commons.apache.org/fileupload">FileUpload</a> library to read the file from the
21   * HTTP request.
22   *
23   * @since Summer 2008
24   */
25  public class FileUploadServlet extends HttpServlet {
26  
27    @Override
28    protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
29      response.setContentType("text/html");
30      PrintWriter pw = response.getWriter();
31      pw.println("<html>");
32      pw.println("<body>");
33  
34      if (JakartaServletFileUpload.isMultipartContent(request)) {
35        DiskFileItemFactory factory = new DiskFileItemFactory.Builder().get();
36        JakartaServletFileUpload<DiskFileItem, DiskFileItemFactory> upload = new JakartaServletFileUpload<>(factory);
37        List<DiskFileItem> items;
38        try {
39          items = upload.parseRequest(request);
40  
41        } catch (FileUploadException ex) {
42          throw new ServletException("Could not upload file", ex);
43        }
44  
45        for (DiskFileItem item : items) {
46          if (!item.isFormField()) {
47            String fileName = item.getName();
48            String contentType = item.getContentType();
49            pw.println("<h1>You uploaded " + fileName + "</h1>");
50            pw.println("<h2>Content type is " + contentType + "</h2>");
51  
52            if (contentType.equals("text/plain")) {
53              pw.println("<pre>");
54  
55              BufferedReader br = new BufferedReader(new InputStreamReader(item.getInputStream()));
56              while (br.ready()) {
57                pw.println(br.readLine());
58              }
59  
60              pw.println("</pre>");
61            }
62          }
63  
64        }
65  
66      } else {
67        pw.println("<h1>You did not upload a file?!</h1>");
68      }
69  
70      ServletInfoServlet.dump(request, pw);
71  
72      pw.println("</body>");
73      pw.println("</html>");
74    }
75  }
