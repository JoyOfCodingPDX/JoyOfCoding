3   import org.w3c.dom.*;

5   /**
6    * This class represents a resident (person) whose first and last
7    * name, as well as whose middle initial, is listing in a phone book.
8    * A <code>Resident</code> is constructed from an XML DOM tree.  If we
9    * were doing this for real, we'd want a way of constructing an empty
10   * <code>Resident</code> and filling in its fields.
11   */
12  public class Resident extends PhoneBookEntry {
13    private String firstName;
14    private String middleInitial;
15    private String lastName;
16    private boolean unlisted = false;
17  
18    /**
19     * Create a new <code>Resident</code> from a <code>Element</code> in
20     * a DOM tree.
21     */
22    public Resident(Element root) {
23      NodeList elements = root.getChildNodes();
24      for (int i = 0; i < elements.getLength(); i++) {
25        Node node = elements.item(i);
26  
27        if (!(node instanceof Element)) {
28          continue;
29        }
30  
31        Element element = (Element) node;
32  
33        switch (element.getNodeName()) {
34          case "first-name": {
35            Node name = element.getFirstChild();
36            Node text = name.getFirstChild();
37            this.firstName = text.getNodeValue();
38            break;
39          }
40  
41          case "initial": {
42            Node text = element.getFirstChild();
43            this.middleInitial = text.getNodeValue();
44            break;
45          }
46  
47          case "last-name": {
48            Node name = element.getFirstChild();
49            Node text = name.getFirstChild();
50            this.lastName = text.getNodeValue();
51            break;
52          }
53  
54          case "address":
55            fillInAddress(element);
56            break;
57  
58          case "phone":
59            fillInPhone(element);
60            break;
61        }
62      }
63  
64      // Check for the "unlisted" attribute
65      if (root.getAttribute("unlisted").equals("true")) {
66        this.unlisted = true;
67      }
68    }
69  
70    public String toString() {
71      return this.firstName + " " + this.middleInitial + " " +
72        this.lastName + "\n" +
73        super.toString();
74    }
75  }
