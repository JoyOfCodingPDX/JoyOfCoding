1   package edu.pdx.cs410J.grader;

3   import javax.mail.Message;
4   import javax.mail.MessagingException;
5   import javax.mail.Session;
6   import javax.mail.internet.InternetAddress;
7   import javax.mail.internet.MimeMessage;
8   import java.util.Properties;
9   
10  public class EmailSender {
11    /**
12     * The name of the SMTP server that is used to send email
13     */
14    protected static String serverName = "mailhost.cs.pdx.edu";
15  
16    private static int emailServerPort = 25;
17  
18    protected static MimeMessage newEmailTo(Session session, String recipient, String subject) throws MessagingException {
19      // Make a new email message
20      MimeMessage message = new MimeMessage(session);
21  
22      InternetAddress[] to = {new InternetAddress(recipient)};
23      message.setRecipients(Message.RecipientType.TO, to);
24      message.setSubject(subject);
25      return message;
26    }
27  
28    protected static Session newEmailSession(boolean debug) {
29      // Obtain a Session for sending email
30      Properties props = new Properties();
31      props.put("mail.smtp.host", serverName);
32      props.put("mail.smtp.port", emailServerPort);
33      props.put("mail.smtp.localhost", "127.0.0.1");
34      Session session = Session.getDefaultInstance(props, null);
35      session.setDebug(debug);
36      return session;
37    }
38  
39    public void setEmailServerPort(int port) {
40      emailServerPort = port;
41    }
42  }
