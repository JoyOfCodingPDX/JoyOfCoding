3   import com.google.inject.Inject;
4   import com.google.inject.Singleton;
5   
6   import javax.swing.*;
7   import java.awt.*;
8   
9   @Singleton
10  public class POASubmissionInformationPanel extends JPanel {
11  
12    @Inject
13    public POASubmissionInformationPanel(POASubmissionInformationWidgets submissionWidgets,
14                                         GradeBookWidget gradeBookWidget,
15                                         StudentsWidget studentsWidget,
16                                         POAAssignmentsWidget assignmentsWidget,
17                                         POAGradeWidgets gradeWidgets) {
18      this.setLayout(new BorderLayout());
19  
20      JPanel information = new JPanel(new GridBagLayout());
21  
22      addComponentOnGrid(submissionWidgets.getSubmitterWidget(), information, 0, 0);
23      addComponentOnGrid(studentsWidget, information, 0, 1);
24  
25      addComponentOnGrid(submissionWidgets.getSubjectWidget(), information, 1, 0);
26      addComponentOnGrid(assignmentsWidget, information, 1, 1);
27  
28      addComponentOnGrid(submissionWidgets.getSubmissionTimeWidget(), information, 2, 0);
29  
30      addComponentOnGrid(gradeBookWidget, information, 3, 0);
31      addComponentOnGrid(gradeWidgets.getGradeWidget(), information, 3, 1);
32  
33      this.add(information, BorderLayout.NORTH);
34      this.add(submissionWidgets.getSubmissionContentWidget(), BorderLayout.CENTER);
35    }
36  
37    private void addComponentOnGrid(JComponent component, JPanel container, int row, int column) {
38      GridBagConstraints constraints = new GridBagConstraints();
39      constraints.fill = GridBagConstraints.HORIZONTAL;
40      constraints.anchor = GridBagConstraints.LINE_START;
41      constraints.gridx = column;
42      constraints.gridy = row;
43  
44      container.add(component, constraints);
45    }
46  
47  }
