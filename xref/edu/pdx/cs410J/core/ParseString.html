5   /**
6    * This classes parses strings from the command line using a
7    * <code>StringTokenizer</code>.
8    */
9   public class ParseString {
10  
11    /**
12     * The second <code>String</code> from the command line contains the
13     * parsing delimiters.
14     */
15    public static void main(String[] args) {
16      String string = args[0];
17      String delimit = args[1];
18      StringTokenizer st;
19      st = new StringTokenizer(string, delimit);
20  
21      while (st.hasMoreTokens()) {
22        String token = st.nextToken();
23        System.out.println("Token: " + token);
24      }
25    }
26  }
