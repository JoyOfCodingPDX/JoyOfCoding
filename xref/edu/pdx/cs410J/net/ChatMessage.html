3   import java.io.Serializable;
4   import java.text.*;
5   import java.util.*;
6   
7   /**
8    * This class represents a message that is passed between two
9    * <code>ChatSession</code>s.
10   */
11  public class ChatMessage implements Serializable {
12  
13    private String sender;   // Name of the sender
14    private Date   date;     // When the messsage was sent
15    private String text;     // Contents of message
16  
17    /**
18     * Creates a new <code>ChatMessage</code> with the current time.
19     */
20    public ChatMessage(String sender, String text) {
21      this.sender = sender;
22      this.date = new Date();
23      this.text = text;
24    }
25  
26    /**
27     * Returns <code>true</code> if this is the last message sent
28     */
29    public boolean isLastMessage() {
30      return this.text.trim().equals("bye");
31    }
32  
33    /**
34     * Returns a textual representation of this <code>ChatMessage</code>
35     * that is suitable for displaying in a <code>ChatSession</code>.
36     */
37    public String toString() {
38      DateFormat df = DateFormat.getTimeInstance(DateFormat.MEDIUM);
39      StringBuffer sb = new StringBuffer();
40      sb.append(this.sender);
41      sb.append(" [");
42      sb.append(df.format(this.date));
43      sb.append("]> ");
44      sb.append(this.text);
45      return sb.toString();
46    }
47  
48  }
