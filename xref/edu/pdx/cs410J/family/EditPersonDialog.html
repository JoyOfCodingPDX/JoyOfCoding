55    /**
56     * Creates a new <code>EditPersonDialog</code> for editing an
57     * existing Person.
58     */
59    public EditPersonDialog(Person person, JFrame owner, 
60                            FamilyTree tree) {
61      this(owner, "Edit Person " + person.getId(), tree);
62      this.person = person;
63  
64      // Fill in information about the person
65      this.idField.setText("" + person.getId());
66      this.idField.setEditable(false);
67  
68      if (this.person.getGender() == Person.FEMALE) {
69        female.setSelected(true);
70        female.setEnabled(true);
71        male.setEnabled(false);
72        male.setSelected(false);
73      }
74      
75      if (this.person.getGender() == Person.MALE) {
76        male.setSelected(true);
77        male.setEnabled(true);
78        female.setSelected(false);
79        female.setEnabled(false);
80      }
81  
82      this.firstNameField.setText(person.getFirstName());
83      this.middleNameField.setText(person.getMiddleName());
84      this.lastNameField.setText(person.getLastName());
85      
86      DateFormat df = DateFormat.getDateInstance(DateFormat.MEDIUM);
87      Date dob = person.getDateOfBirth();
88      if (dob != null) {
89        this.dobField.setText(df.format(dob));
90      }
91      Date dod = person.getDateOfDeath();
92      if (dod != null) {
93        this.dodField.setText(df.format(dod));
94      }
95  
96      this.father = person.getFather();
97      if (this.father != null) {
98        this.fatherField.setText(this.father.getFullName());
99      }
100 
101     this.mother = person.getMother();
102     if (this.mother != null) {
103       this.motherField.setText(this.mother.getFullName());
104     }
105   }

236             id = Integer.parseInt(idField.getText());

260               // Parse error

262             }
263           }
264           
265           Gender gender;
266           if (group.getSelection().getActionCommand().equals("male")) {
267             gender = Person.MALE;
268 
269           } else {
270             gender = Person.FEMALE;
271           }
272  
273           // Okay, everything parsed alright
274           if (person == null) {
275             person = new Person(id, gender);
276           }
277 
278           person.setFirstName(firstNameField.getText());

282           person.setDateOfDeath(dod);

286           }
287 
288           if (father != null) {
289             person.setFather(father);
290           }
291

298             }

334   /**
335    * Tries very, very hard to parse the a date.  We assume that the
336    * text is neither empty nor <code>null</code>.
337    */
338   private Date parseDate(String text) {
339     DateFormat formats[] = new DateFormat[] {
340       DateFormat.getDateInstance(DateFormat.SHORT),
341       DateFormat.getDateInstance(DateFormat.MEDIUM),
342       DateFormat.getDateInstance(DateFormat.LONG),
343       DateFormat.getDateInstance(DateFormat.FULL),
344     };
345 
346     for (int i = 0; i < formats.length; i++) {
347       DateFormat df = formats[i];
348       try {
349         Date date = df.parse(text);
350         return date;
351 
352       } catch (ParseException ex) {
353         continue;
354       }
355     }
356 
357     error("Could not parse date: " + text);
358     return null;
359   }
