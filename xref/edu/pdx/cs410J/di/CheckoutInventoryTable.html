14  /**
15   * A table that shows the inventory of books and lets the user select one to be added to a shopping cart.  When a book
16   * is selected, a {@link TableModelEvent} is fired to listeners on this table.
17   */
18  class CheckoutInventoryTable extends JTable
19  {
20  
21      public CheckoutInventoryTable( CheckoutInventoryTableModel model )
22      {
23          super(model);
24          TableColumn column = this.getColumnModel().getColumn( model.getColumnCount() - 1 );
25          column.setCellEditor( new TableButton( model ) );
26          column.setCellRenderer( new TableButton( model ) );
27      }
28  
29      /**
30       * Renders a button that
31       */
32      private static class TableButton implements TableCellRenderer, TableCellEditor
33      {
34          private final BookInventoryTableModel model;
35  
36          public TableButton( BookInventoryTableModel model )
37          {
38              this.model = model;
39          }
40  
41          public Component getTableCellRendererComponent( JTable table, Object value,  boolean isSelected, boolean hasFocus, int row, int column )
42          {
43              return getButton( row );
44          }
45  
46          private Component getButton( final int row )
47          {
48              JButton button = new JButton( "Add" );
49              button.addActionListener( new ActionListener() {
50  
51                  public void actionPerformed( ActionEvent event )
52                  {
53                     model.fireTableChanged( new TableModelEvent( model, row, row, model.getColumnCount() -1 ) );
54                  }
55              });
56              return button;
57          }
58  
59          public Component getTableCellEditorComponent( JTable table, Object value, boolean isSelected, int row, int column )
60          {
61              return getButton(row);
62          }
63  
64          public Object getCellEditorValue()
65          {
66              return null;
67          }
68  
69          public boolean isCellEditable( EventObject eventObject )
70          {
71              return true;
72          }
73  
74          public boolean shouldSelectCell( EventObject eventObject )
75          {
76              return true;
77          }
78  
79          public boolean stopCellEditing()
80          {
81              return true;
82          }
83  
84          public void cancelCellEditing()
85          {
86  
87          }
88  
89          public void addCellEditorListener( CellEditorListener cellEditorListener )
90          {
91  
92          }
93  
94          public void removeCellEditorListener( CellEditorListener cellEditorListener )
95          {
96  
97          }
98      }
