6   import javax.servlet.*;
7   import javax.servlet.http.HttpServletRequest;
8   import java.io.IOException;
9   import java.util.logging.Logger;
10  
11  /**
12   * Logs all calls to the rest services
13   */
14  @Singleton
15  public class RestLoggingFilter implements Filter
16  {
17      private final Logger logger;
18  
19      private final CreditCardDatabase cards;
20  
21      @Inject
22      public RestLoggingFilter(Logger logger, CreditCardDatabase cards) {
23        this.logger = logger;
24        this.cards = cards;        
25      }
26  
27      public void init( FilterConfig config ) throws ServletException
28      {
29        // Initialize some credit cards for testing purposes.  This probably isn't the best place to do this. 
30        for (int i = 1; i <= 10; i++) {
31            CreditCard card = new CreditCard(String.valueOf(i));
32            cards.setBalance( card, 100.0 );
33        }
34      }
35  
36      public void doFilter( ServletRequest request, ServletResponse response, FilterChain chain )
37          throws IOException, ServletException
38      {
39          long begin = System.currentTimeMillis();
40  
41          StringBuilder description = new StringBuilder("Request from ");
42          description.append(request.getRemoteAddr()).append(":").append(request.getRemotePort());
43  
44          if (request instanceof HttpServletRequest) {
45            HttpServletRequest http = (HttpServletRequest) request;
46  
47            description.append(" for ").append( http.getMethod() ).append(" ").append(http.getRequestURI());
48          }
49          logger.info("Begin " + description);
50  
51          chain.doFilter( request, response );
52  
53          long delta = System.currentTimeMillis() - begin;
54  
55          logger.info("End (" + delta + " ms) " + description);
56  
57      }
58  
59      public void destroy()
60      {
61  
62      }
63  }
