3   import javax.xml.bind.JAXBContext;
4   import javax.xml.bind.JAXBException;
5   import javax.xml.bind.Marshaller;
6   import javax.xml.bind.Unmarshaller;
7   import java.io.File;
8   import java.io.IOException;
9   
10  /**
11   * The superclass of classes that store objects serialized to a file using JAXB
12   */
13  public abstract class JaxbDatabase
14  {
15      private final File directory;
16      private final String fileName;
17      private final JAXBContext xmlContext;
18      private final File file;
19  
20      protected JaxbDatabase( File directory, String fileName, Class<?>... jaxbClasses )
21          throws IOException, JAXBException
22      {
23          this.directory = directory;
24          this.fileName = fileName;
25          this.xmlContext = JAXBContext.newInstance( jaxbClasses );
26  
27          this.directory.mkdirs();
28          if (!this.directory.exists()) {
29            throw new IOException( "Could not create data directory: " + this.directory);
30          }
31  
32  
33          this.file = new File(this.directory, this.fileName);
34      }
35  
36      /**
37       * Writes an object as XML to the data file
38       * @param xml The object to marshal
39       */
40      protected void writeXml( Object xml )
41      {
42          try
43          {
44              Marshaller marshaller = this.xmlContext.createMarshaller();
45              marshaller.marshal( xml, new File(this.directory, this.fileName) );
46          }
47          catch ( JAXBException ex )
48          {
49              throw new IllegalStateException( "Could not save inventory", ex);
50          }
51      }
52  
53      public File getDatabaseFile() {
54          return file;
55      }
56  
57      /**
58       * Read the XML data file and returns the unmarshalled object
59       * @return The object in the XML file or null if the file doesn't exist
60       * @throws JAXBException if we can't read the file
61       */
62      protected Object readFile()
63          throws JAXBException
64      {
65          System.out.println("Reading xml data from " + file );
66  
67          if ( this.file.exists()) {
68            Unmarshaller unmarshaller = this.xmlContext.createUnmarshaller();
69            return unmarshaller.unmarshal( this.file );
70          } else {
71            return null;
72          }
73      }
74  }
