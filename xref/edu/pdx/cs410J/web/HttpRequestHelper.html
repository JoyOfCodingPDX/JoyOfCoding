23     * @return A <code>Response</code> summarizing the result of the GET
24     */
25    protected Response get(String urlString, Map<String, String> parameters) throws IOException {
26      StringBuilder query = encodeParameters(parameters);
27      if (query.length() > 0) {
28        query.insert(0, '?');
29      }
30  
31      URL url = new URL(urlString + query);
32      HttpURLConnection conn = (HttpURLConnection) url.openConnection();
33      conn.setRequestMethod("GET");
34      conn.setDoOutput(true);
35      conn.setDoInput(true);
36  
37      return new Response(conn);
38  
39    }

61    private Response sendEncodedRequest(String urlString, String requestMethod, Map<String, String> parameters) throws IOException {
62      StringBuilder data = encodeParameters(parameters);
63  
64      URL url = new URL(urlString);
65      HttpURLConnection conn = (HttpURLConnection) url.openConnection();
66      conn.setRequestMethod(requestMethod);
67      conn.setDoOutput(true);
68  
69      OutputStreamWriter wr = new OutputStreamWriter(conn.getOutputStream(), Charset.forName("UTF-8"));
70      wr.write(data.toString());
71      wr.flush();
72  
73      Response response = new Response(conn);
74      wr.close();
75  
76      return response;
77    }

79    /**
80     * Encodes parameters to be sent to the server via an HTTP GET, POST, or DELETE
81     * @param parameters The parameter key/value pairs
82     * @return The encoded parameters
83     * @throws java.io.UnsupportedEncodingException If we can't encode UTF-8
84     */
85    private StringBuilder encodeParameters(Map<String, String> parameters) throws UnsupportedEncodingException {
86      StringBuilder query = new StringBuilder();
87      for (Iterator<Map.Entry<String, String>> iter = parameters.entrySet().iterator(); iter.hasNext(); ) {
88        Map.Entry<String, String> pair = iter.next();
89        String key = pair.getKey();
90        String value = pair.getValue();
91        query.append(URLEncoder.encode(key, StandardCharsets.UTF_8));
92        query.append("=");
93        query.append(URLEncoder.encode(value, StandardCharsets.UTF_8));
94        if (iter.hasNext()) {
95          query.append("&");
96        }
97      }
98      return query;
99    }

144     private int contentLines = 0;

146     private Response(HttpURLConnection conn) throws IOException {
147       this.code = conn.getResponseCode();
148       if (this.code != java.net.HttpURLConnection.HTTP_OK) {
149         this.content = conn.getResponseMessage();
150         return;
151       }
152 
153       StringBuilder content = new StringBuilder();
154       BufferedReader rd = new BufferedReader(new InputStreamReader(conn.getInputStream()));
155       String line;
156       while ((line = rd.readLine()) != null) {
157         content.append(line);
158         content.append("\n");
159         contentLines++;
160       }
161       rd.close();
162 
163       this.content = content.toString().trim();
164     }
165 
166     /**
167      * Returns the HTTP status code of the response
168      * @see java.net.HttpURLConnection#getResponseCode()
169      */
170     public int getCode() {
171       return code;
172     }

174     /**
175      * Returns the (presumably textual) response from the URL

177     public String getContent() {

179     }

182      * Returns the number of lines in the response's content
