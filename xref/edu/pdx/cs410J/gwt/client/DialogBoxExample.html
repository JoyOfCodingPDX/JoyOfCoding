3   import com.google.gwt.user.client.ui.*;
4   import com.google.gwt.event.dom.client.ClickHandler;
5   import com.google.gwt.event.dom.client.ClickEvent;
6   
7   /**
8    * Pops up a GWT {@link DialogBox}
9    */
10  public class DialogBoxExample extends Example {
11    public DialogBoxExample() {
12      super("Dialog Box");
13  
14      Button autoHide = new Button("Auto-hide dialog");
15      autoHide.addClickHandler(new ClickHandler() {
16          public void onClick( ClickEvent clickEvent )
17          {
18              createDialogBox(true, false).show();
19          }
20      });
21      add(autoHide);
22  
23      Button modal = new Button("Modal dialog");
24      modal.addClickHandler(new ClickHandler() {
25        public void onClick(ClickEvent event) {
26          createDialogBox(false, true).show();
27        }
28      });
29      add(modal);
30    }
31  
32    private DialogBox createDialogBox(boolean autoHide, boolean modal) {
33      final DialogBox box = new DialogBox(autoHide, modal);
34      box.setText("Auto-hide dialog");
35  
36      DockPanel panel = new DockPanel();
37      panel.add(new Label("Click outside me"), DockPanel.CENTER);
38      panel.add(new Button("Close", new ClickHandler() {
39        public void onClick(ClickEvent event) {
40          box.hide();
41        }
42      }), DockPanel.SOUTH);
43      box.setWidget(panel);
44      box.center();
45      return box;
46    }
47  }
