19    /**
20     * Creates a new <code>Student</code>                                             
21     *                                                                                
22     * @param name                                                                    
23     *        The student's name                                                      
24     * @param classes                                                                 
25     *        The names of the classes the student is taking.  A student              
26     *        may take zero or more classes.                                          
27     * @param gpa                                                                     
28     *        The student's grade point average                                       
29     * @param gender                                                                  
30     *        The student's gender ("male" or "female", or "other", case insensitive)
31     * @throws IllegalArgumentException
32     *         If the GPA is not between 0.0 and 4.0 or if the gender is unknown
33     */
34    public Student(String name, Set<String> classes, double gpa, String gender) throws IllegalArgumentException {
35      super(name);
36  
37      if (gpa > 4.0) {
38        throw new IllegalArgumentException(gpa + " is too big");
39  
40      } else if (gpa < 0.0) {
41        throw new IllegalArgumentException(gpa + " is too small");
42      }
43  
44      this.classes = classes;
45  
46  
47      this.gpa = gpa;
48  
49      if (gender.equalsIgnoreCase("female")) {
50        this.genderPronoun = "She";
51  
52      } else if (gender.equalsIgnoreCase("male")) {
53        this.genderPronoun = "He";
54  
55      } else if (gender.equalsIgnoreCase("other")) {
56        this.genderPronoun = "They";
57  
58      } else {
59        throw new IllegalArgumentException("Unknown gender: " + gender);
60      }
61    }
62  
63    /**
64     * All students say "This class is too much work"

66    @Override

68      return "This class is too much work";
69    }

71    double getGpa() {
72      return this.gpa;

75    /**
76     * Returns a <code>String</code> that describes this
77     * <code>Student</code>.
78     */
79    @Override
80    public String toString() {
81      int numberOfClasses = this.classes.size();
82      StringBuilder sb = new StringBuilder()
83        .append(getName()).append(" has a GPA of ").append(getGpa())
84        .append(" and is taking ").append(numberOfClasses).append(" class");
85      appendNumberOfClassesSuffix(numberOfClasses, sb);
86  
87      appendClassNames(sb);
88  
89      sb.append("  ");
90      sb.append(getGenderPronoun());
91      sb.append(" says \"");
92      sb.append(says());
93      sb.append("\".");
94  
95      return sb.toString();
96    }

98    private void appendClassNames(StringBuilder sb) {
99      int numberOfClasses = this.classes.size();
100     if (numberOfClasses == 1) {
101       sb.append(this.classes.iterator().next());
102 
103     } else if (numberOfClasses == 2) {
104       Iterator<String> iter = this.classes.iterator();
105       sb.append(iter.next());
106       sb.append(" and ");
107       sb.append(iter.next());
108 
109     } else {
110       for (Iterator<String> iter = this.classes.iterator(); iter.hasNext(); ) {
111         String className = iter.next();
112         if (!iter.hasNext()) {
113           sb.append("and ");
114         }
115         sb.append(className);
116         if (iter.hasNext()) {
117           sb.append(", ");
118         }
119       }
120     }
121 
122 
123     sb.append(".");
124   }

126   private void appendNumberOfClassesSuffix(int numberOfClasses, StringBuilder sb) {
127     switch (numberOfClasses) {
128       case 0:
129         sb.append("es.");
130         break;
131       case 1:
132         sb.append(": ");
133         break;
134       default:
135         sb.append("es: ");
136         break;
137     }
138   }
