019  /**
020   * Converts a FamilyTree to text and returns the text as a String.
021   */
022  @Override
023  protected String getStringFor(FamilyTree tree) {
024    StringWriter sw = new StringWriter();
025    PrintWriter pw = new PrintWriter(sw, true);
026    Dumper dumper = new TextDumper(pw);
027    dumper.dump(tree);
028    String s = sw.toString();
029    if (Boolean.getBoolean("TextTest.DUMP_TEXT")) {
030      System.out.println(s);
031    }
032    return s;
033  }
034
035  /**
036   * Parsers a FamilyTree from a String containing XML
037   */
038  @Override
039  protected FamilyTree getFamilyTreeFor(String s)
040    throws FamilyTreeException {
041
042    // Parse the XML from the String
043    StringReader sr = new StringReader(s);
044    Parser parser = new TextParser(sr);
045    return parser.parse();
046  }
