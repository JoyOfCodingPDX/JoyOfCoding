047  private File makeFileWithPath(String... path) {
048    File dir = new File(System.getProperty("user.dir"));
049
050    for (String part : path) {
051      dir = new File(dir, part);
052    }
053
054    return dir;
055  }

078
079    LocalDateTime actual = Submit.ManifestAttributes.parseSubmissionTime(legacySubmissionTime);
