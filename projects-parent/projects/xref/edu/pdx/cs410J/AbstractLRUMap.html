5   /**
6    * This is the abstract superclass of an "LRU Map".  An LRU Map is a
7    * Java {@link Map} that has a bounded number of mappings.  If a new
8    * mapping is added to an LRU Map that already has the maximum number
9    * of mappings, then the least recently used mapping is removed.  An
10   * element is used when it is added to or gotten from the Map.
11   */
12  public abstract class AbstractLRUMap<K, V> extends AbstractMap<K, V> {
13  
14    /** The maximum number of mappings that this map can hold */
15    protected int capacity;
16  
17    ///////////////////////  Constructors  ///////////////////////
18  
19    /**
20     * Creates a new LRU Map that will hold a given number of mappings
21     *
22     * @param capacity The maximum number of mappings in this map
23     * @throws IllegalArgumentException
24     *         <code>capacity</code> is negative
25     */
26    protected AbstractLRUMap(int capacity) {
27      if (capacity < 0) {
28        String s = "Max mappings must be greater than 0";
29        throw new IllegalArgumentException(s);
30      }
31  
32      this.capacity = capacity;
33    }
34  
35    /**
36     * Returns the names of the students who implemented this LRU Map.
37     */
38    public abstract List<String> getStudentNames();
39  
40    /**
41     * When a mapping is made in an LRU that already contains the
42     * maximum number of mappings, the Least Recently Used element is
43     * removed.
44     */
45    public abstract V put(K key, V value);
46  
47    /**
48     * Getting an element from a map marks the mapping as "used".  Note
49     * that the type of <code>key</code> must be <code>Object</code>
50     * so that, after type erasure, it will be compatible with the
51     * pre-generic API.
52     */
53    public abstract V get(Object key);
54  
55  	/**
56  	 * Removes the given key from this map.  Note that the type of
57  	 * <code>key</code> must be <code>Object</code> so that, after type
58  	 * erasure, it will be compatible with the pre-generic API.
59  	 */
60    public abstract V remove(Object key);
61  
62    public void putAll(Map<? extends K, ? extends V> map) {
63      for (K key : map.keySet()) {
64        V value = map.get(key);
65        this.put(key, value);
66      }

71  //       Map.Entry<? extends K, ? extends V> entry = iter.next();
72  //       this.put(entry.getKey(), entry.getValue());
73  //     }
74    }
75  
76    public abstract void clear();
77  
78  }
