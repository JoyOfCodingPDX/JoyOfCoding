3   import edu.pdx.cs410J.AppointmentBookParser;
4   import edu.pdx.cs410J.ParserException;
5   
6   import java.io.BufferedReader;
7   import java.io.FileReader;
8   import java.io.IOException;
9   import java.io.Reader;
10  
11  public class TextParser implements AppointmentBookParser<AppointmentBook> {
12    private final Reader reader;
13  
14    public TextParser(Reader reader) {
15      this.reader = reader;
16    }
17  
18    @Override
19    public AppointmentBook parse() throws ParserException {
20      try (
21        BufferedReader br = new BufferedReader(this.reader)
22      ) {
23  
24        String owner = br.readLine();
25  
26        if (owner == null) {
27          throw new ParserException("Missing owner");
28        }
29  
30        return new AppointmentBook(owner);
31  
32      } catch (IOException e) {
33        throw new ParserException("While parsing appointment book text", e);
34      }
35    }
36  }
