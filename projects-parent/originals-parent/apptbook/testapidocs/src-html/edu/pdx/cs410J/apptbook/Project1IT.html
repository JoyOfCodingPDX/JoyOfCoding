001package edu.pdx.cs410J.apptbook;
002
003import edu.pdx.cs410J.InvokeMainTestCase;
004import org.junit.Test;
005
006import static org.hamcrest.CoreMatchers.containsString;
007import static org.hamcrest.CoreMatchers.equalTo;
008import static org.hamcrest.MatcherAssert.assertThat;
009
010/**
011 * Integration tests for the {@link Project1} main class.
012 */
013public class Project1IT extends InvokeMainTestCase {
014
015  /**
016   * Invokes the main method of {@link Project1} with the given arguments.
017   */
018  private MainMethodResult invokeMain(String... args) {
019    return invokeMain( Project1.class, args );
020  }
021
022  /**
023   * Tests that invoking the main method with no arguments issues an error
024   */
025  @Test
026  public void testNoCommandLineArguments() {
027    MainMethodResult result = invokeMain();
028    assertThat(result.getExitCode(), equalTo(1));
029    assertThat(result.getTextWrittenToStandardError(), containsString("Missing command line arguments"));
030  }
031
032}
