15    @Test
16    void appointmentBookOwnerIsDumpedInTextFormat() {
17      String owner = "Test Appointment Book";
18      AppointmentBook book = new AppointmentBook(owner);
19  
20      StringWriter sw = new StringWriter();
21      TextDumper dumper = new TextDumper(sw);
22      dumper.dump(book);
23  
24      String text = sw.toString();
25      assertThat(text, containsString(owner));
26    }
27  
28    @Test
29    void canParseTextWrittenByTextDumper(@TempDir File tempDir) throws IOException, ParserException {
30      String owner = "Test Appointment Book";
31      AppointmentBook book = new AppointmentBook(owner);
32  
33      File textFile = new File(tempDir, "apptbook.txt");
34      TextDumper dumper = new TextDumper(new FileWriter(textFile));
35      dumper.dump(book);
36  
37      TextParser parser = new TextParser(new FileReader(textFile));
38      AppointmentBook read = parser.parse();
39      assertThat(read.getOwnerName(), equalTo(owner));
40    }
