001package edu.pdx.cs410J.apptbookweb;
002
003import com.google.common.annotations.VisibleForTesting;
004import edu.pdx.cs410J.web.HttpRequestHelper;
005
006import java.io.IOException;
007import java.util.Map;
008
009import static java.net.HttpURLConnection.HTTP_OK;
010
011/**
012 * A helper class for accessing the rest client.  Note that this class provides
013 * an example of how to make gets and posts to a URL.  You'll need to change it
014 * to do something other than just send dictionary entries.
015 */
016public class AppointmentBookRestClient extends HttpRequestHelper {
017  private static final String WEB_APP = "apptbook";
018  private static final String SERVLET = "appointments";
019
020  private final String url;
021
022
023  /**
024   * Creates a client to the appointment book REST service running on the given host and port
025   *
026   * @param hostName The name of the host
027   * @param port     The port
028   */
029  public AppointmentBookRestClient(String hostName, int port) {
030    this.url = String.format("http://%s:%d/%s/%s", hostName, port, WEB_APP, SERVLET);
031  }
032
033  /**
034   * Returns all dictionary entries from the server
035   */
036  public Map<String, String> getAllDictionaryEntries() throws IOException {
037    Response response = get(this.url, Map.of());
038    return Messages.parseDictionary(response.getContent());
039  }
040
041  /**
042   * Returns the definition for the given word
043   */
044  public String getDefinition(String word) throws IOException {
045    Response response = get(this.url, Map.of("word", word));
046    throwExceptionIfNotOkayHttpStatus(response);
047    String content = response.getContent();
048    return Messages.parseDictionaryEntry(content).getValue();
049  }
050
051  public void addDictionaryEntry(String word, String definition) throws IOException {
052    Response response = postToMyURL(Map.of("word", word, "definition", definition));
053    throwExceptionIfNotOkayHttpStatus(response);
054  }
055
056  @VisibleForTesting
057  Response postToMyURL(Map<String, String> dictionaryEntries) throws IOException {
058    return post(this.url, dictionaryEntries);
059  }
060
061  public void removeAllDictionaryEntries() throws IOException {
062    Response response = delete(this.url, Map.of());
063    throwExceptionIfNotOkayHttpStatus(response);
064  }
065
066  private Response throwExceptionIfNotOkayHttpStatus(Response response) {
067    int code = response.getCode();
068    if (code != HTTP_OK) {
069      throw new AppointmentBookRestException(code);
070    }
071    return response;
072  }
073
074  @VisibleForTesting
075  class AppointmentBookRestException extends RuntimeException {
076    public AppointmentBookRestException(int httpStatusCode) {
077      super("Got an HTTP Status Code of " + httpStatusCode);
078    }
079  }
080}
